082    @SuppressWarnings("rawtypes")
083    public static Object convert(ParsedValue pv) {
084        Object value = null;
085        if (pv == null) {
086            return null;
087        }
088
089        if (pv.getConverter() != null) {
090            try {
091                @SuppressWarnings("unchecked")
092                Object converted = pv.getConverter().convert(pv, null);
093                value = converted;
094            } catch (RuntimeException ex) {
095                // OK, can't be resolved, a lookup possibly
096            }
097        } else {
098            value = pv.getValue();
099        }
100        if (value instanceof ParsedValue) {
101            value = convert((ParsedValue) value);
102        }
103        return value;
104    }

123    // Retrieve the value for a sub property.
124    public static Object getSubPropertyValue(String property, Object value) {
125        if (value instanceof Collection) {
126            Collection<?> values = (Collection<?>) value;
127            List<Object> subValues = new ArrayList<>();
128            for (Object bf : values) {
129                subValues.add(getSubPropertyValue(property, bf));
130            }
131            return subValues;
132        } else if (value != null && value.getClass().isArray()) {
133            Object newArray = Array.newInstance(value.getClass().getComponentType(), Array.getLength(value));
134            for (int i = 0; i < Array.getLength(value); i++) {
135                Array.set(newArray, i, getSubPropertyValue(property, Array.get(value, i)));
136            }
137            return newArray;
138
139            //
140            // Background
141            //
142        } else if (value instanceof Background) {
143            Background background = (Background) value;
144            if (background.getFills() != null) {
145                return getSubPropertyValue(property, background.getFills());
146            } else if (background.getImages() != null) {
147                return getSubPropertyValue(property, background.getImages());
148            }
149        } else if (value instanceof BackgroundFill) {
150            return subBackgroundFill(property, (BackgroundFill) value);
151        } else if (value instanceof BackgroundImage) {
152            return subBackgroundImage(property, (BackgroundImage) value);
153
154            //
155            // Border
156            //
157        } else if (value instanceof Border) {
158            Border border = (Border) value;
159            if (border.getStrokes() != null) {
160                return getSubPropertyValue(property, border.getStrokes());
161            } else if (border.getImages() != null) {
162                return getSubPropertyValue(property, border.getImages());
163            }
164        } else if (value instanceof BorderStroke) {
165            return subBorderStroke(property, (BorderStroke) value);
166        } else if (value instanceof BorderImage) {
167            return subBorderImage(property, (BorderImage) value);
168
169            //
170            // Font
171            //
172        } else if (value instanceof Font) {
173            return subFont(property, (Font) value);
174        }
175        return getValue(property, null, value);
176    }

191    // FX to String value transformation entry point.
192    private static String getValue(String property, Rule r, Object eventValue) throws IllegalArgumentException {
193        if (r == null) {
194            return format(property, retrieveValue(property, eventValue));
195        }
196
197        for (Declaration d : r.getDeclarations()) {
198            if (d.getProperty().equals(property)) {
199                if (property.equals("-fx-background-radius") || property.equals("-fx-border-radius")) { //NOI18N
200                    return format(property, getRadiusCssString(property, d.getParsedValue()));
201                } else {
202                    return format(property, getCssString(property, d.getParsedValue()));
203                }
204            }
205        }
206        throw new IllegalArgumentException("Can't compute a value");//NOI18N
207    }

226    // The difference between retrieveValue and getCssStringValue
227    // is that the ParsedValue is converted in the retrieveValue case, and is not converted in the 
228    // getCssStringValue
229    // When converting, we loose the CSS textual format present in the CSS source,
230    // for instance the lookup information, or 'em' unit.
231    @SuppressWarnings("rawtypes")
232    private static String getCssString(String property, ParsedValue value) {
233
234        // TODO : this method should be rewritten in a cleaner way...
235        if (value == null) {
236            return "null"; //NOI18N
237        }
238
239        // I don't like that but this is needed to only have a conversion for gradient.
240        // The gradient.toString is much better than the ParsedValue.toString.
241        if (value.getConverter() instanceof LinearGradientConverter
242                || value.getConverter() instanceof PaintConverter.RadialGradientConverter) {
243
244            try {
245                @SuppressWarnings("unchecked")//NOI18N
246                Object converted = value.getConverter().convert(value, null);
247
248                return toCssString(converted);
249            } catch (RuntimeException ex) {
250            }
251        }
252
253        Object obj = value.getValue();
254        if (obj instanceof ParsedValue) {
255            return getCssString(property, (ParsedValue) obj);
256        }
257        StringBuilder builder = new StringBuilder();
258        boolean isDerive = value.getConverter() instanceof DeriveColorConverter || value.getConverter() instanceof DeriveSizeConverter;
259        boolean isLadder = value.getConverter() instanceof LadderConverter;
260        if (isDerive) {
261            builder.append("derive("); //NOI18N
262        }
263        if (isLadder) {
264            builder.append("ladder("); //NOI18N
265        }
266        if (obj instanceof ParsedValue[]) {
267            ParsedValue[] array = (ParsedValue[]) obj;
268            boolean isArrayValue = false;
269            if (array.length >= 1) {
270                ParsedValue pval = array[0];
271                Object val = null;
272                if (pval != null) {
273                    if (pval.getConverter() instanceof LinearGradientConverter
274                            || pval.getConverter() instanceof PaintConverter.RadialGradientConverter) {
275                        val = null;
276                    } else {
277                        val = pval.getValue();
278                    }
279                }
280                isArrayValue = val != null && val.getClass().isArray();
281            }
282            boolean singleForEquality = singleForEquality(property) && !isArrayValue;
283            StringBuilder b = new StringBuilder();
284            if (singleForEquality) {
285                String latest = null;
286                boolean areEquals = true;
287                List<String> values = new ArrayList<>(array.length);
288                for (ParsedValue v : array) {
289                    String current = getCssString(property, v);
290                    values.add(current);
291                    areEquals &= (latest == null || current.equals(latest));
292                    latest = current;
293                }
294                if (areEquals) {
295                    String val = values.get(0);
296                    val = removeDotZeroPxPercent(val);
297                    b.append(val);
298                } else {
299                    for (int i = 0; i < values.size(); i++) {
300                        b.append(values.get(i));
301                        if (i < array.length - 1) {
302                            b.append(" "); //NOI18N
303                        }
304                    }
305                }
306            } else {
307                for (int i = 0; i < array.length; i++) {
308                    ParsedValue v = array[i];
309                    String val = getCssString(property, v);
310                    val = removeDotZeroPxPercent(val);
311                    b.append(val);
312                    if ((i < array.length - 1) && val.length() > 0) {
313                        b.append(", "); //NOI18N
314                    }
315                }
316            }
317
318            builder.append(b.toString());
319        } else {
320            if (obj instanceof ParsedValue[][]) {
321                ParsedValue[][] arr = (ParsedValue[][]) obj;
322                for (int i = 0; i < arr.length; i++) {
323                    String val = retrieveValue(property, arr[i]);
324                    builder.append(val);
325                    if ((i < arr.length - 1) && val.length() > 0) {
326                        builder.append(", "); //NOI18N
327                    }
328                }
329            } else {
330                builder.append(retrieveValue(property, obj));
331            }
332        }
333        if (isDerive || isLadder) {
334            builder.append(")"); //NOI18N
335        }
336        return builder.toString();
337    }

339    @SuppressWarnings("rawtypes")
340    private static String getRadiusCssString(String property, ParsedValue value) {
341        // TODO : Ideally should be included in the generic getCssString() method
342        
343        // See  http://www.w3.org/TR/css3-background/#the-border-radius 
344        
345        assert property.equals("-fx-background-radius") || property.equals("-fx-border-radius"); //NOI18N
346        StringBuilder sbAll = new StringBuilder();
347        Object obj = value.getValue();
348        if (!(obj instanceof ParsedValue[])) {
349            return null;
350        }
351        ParsedValue[] pvArray = (ParsedValue[]) obj;
352        int index = 0;
353        for (ParsedValue pvItem : pvArray) {
354            // We have a CornerRadii representation here:
355            // double dimension array:
356            // 1- horizontal/vertical radii
357            // 2- value per corner
358            // If all the values are identical, only a single value is used.
359            obj = pvItem.getValue();
360            if (!(obj instanceof ParsedValue[][])) {
361                return null;
362            }
363            ParsedValue[][] pvArray2 = (ParsedValue[][]) obj;
364            StringBuilder sbCornerRadii = new StringBuilder();
365            Size initSize = null;
366            boolean areEquals = true;
367            int index2 = 0;
368            for (ParsedValue[] pvArray1 : pvArray2) {
369                // horizontal or vertical list 
370                for (ParsedValue pvItem2 : pvArray1) {
371                    obj = pvItem2.getValue();
372                    if (!(obj instanceof Size)) {
373                        return null;
374                    }
375                    Size size = (Size) obj;
376                    sbCornerRadii.append(size).append(" "); //NOI18N
377                    if (initSize == null) {
378                        initSize = size;
379                    } else if (!initSize.equals(size)) {
380                        areEquals = false;
381                    }
382                }
383                if (index2 != pvArray2.length - 1) {
384                    // Separator between the horizontal / vertical lists
385                    sbCornerRadii.append(" / "); //NOI18N
386                }
387                index2++;
388            }
389            if (areEquals) {
390                sbAll.append(initSize);
391            } else {
392                sbAll.append(sbCornerRadii.toString().trim());
393            }
394            if (index != pvArray.length - 1) {
395                sbAll.append(", "); //NOI18N
396            }
397            index++;
398        }
399        return removeDotZeroPxPercent(sbAll.toString());
400    }

402    private static String retrieveValue(String property, Object eventValue) {
403        if (eventValue instanceof ParsedValue) {
404            eventValue = convert((ParsedValue) eventValue);
405        }
406
407        if (eventValue == null) {
408            return "null"; //NOI18N
409        }
410        StringBuilder builder = new StringBuilder();
411        if (eventValue instanceof List) {
412            List<?> values = (List<?>) eventValue;
413            int length = values.size();
414            for (int i = 0; i < length; i++) {
415                String val = retrieveValue(property, values.get(i));
416                builder.append(val);
417                if ((i < length - 1) && val.length() > 0) {
418                    builder.append(", "); //NOI18N
419                }
420            }
421        } else if (eventValue.getClass().isArray()) {
422            int length = Array.getLength(eventValue);
423            for (int i = 0; i < length; i++) {
424                String val = retrieveValue(property, Array.get(eventValue, i));
425                builder.append(val);
426                if ((i < length - 1) && val.length() > 0) {
427                    builder.append(", "); //NOI18N
428                }
429            }
430        } else if (eventValue instanceof Background) {
431            Background background = (Background) eventValue;
432            if (background.getFills() != null) {
433                return retrieveValue(property, background.getFills());
434            } else if (background.getImages() != null) {
435                return retrieveValue(property, background.getImages());
436            }
437        } else if (eventValue instanceof Border) {
438            Border border = (Border) eventValue;
439            if (border.getStrokes() != null) {
440                return retrieveValue(property, border.getStrokes());
441            } else if (border.getImages() != null) {
442                return retrieveValue(property, border.getImages());
443            }
444        } else if (eventValue instanceof BackgroundFill) {
445            builder.append(backgroundFillToString(property, (BackgroundFill) eventValue));
446        } else if (eventValue instanceof CornerRadii) {
447            builder.append(cornerRadiiToString(property, (CornerRadii) eventValue));
448        } else if (eventValue instanceof BackgroundImage) {
449            builder.append(backgroundImageToString(property, (BackgroundImage) eventValue));
450        } else if (eventValue instanceof BorderStroke) {
451            builder.append(borderStrokeToString(property, (BorderStroke) eventValue));
452        } else if (eventValue instanceof BorderImage) {
453            builder.append(borderImageToString(property, (BorderImage) eventValue));
454        } else if (eventValue instanceof Font) {
455            builder.append(fontToString(property, (Font) eventValue));
456        } else if (eventValue instanceof Paint) {
457            builder.append(paintToString((Paint) eventValue).toLowerCase(Locale.ROOT));
458        } else if (eventValue instanceof Insets) {
459            builder.append(insetsValue((Insets) eventValue));
460        } else if (eventValue instanceof Effect) {
461            builder.append(effectValue((Effect) eventValue));
462        } else {
463            String str = EditorUtils.valAsStr(eventValue);
464            if (str == null) {
465                str = "null"; //NOI18N
466            } else {
467                str = str.replaceAll("\n", " ");//NOI18N
468                // Remove memory address if any
469                str = str.split("@")[0]; //NOI18N
470                str = removeDotZeroPxPercent(str);
471            }
472            builder.append(str);
473        }
474        return builder.toString();
475    }

477    private static String getColorAsWebString(Color c) {
478        int red = (int) Math.round(c.getRed() * 255.0);
479        int green = (int) Math.round(c.getGreen() * 255.0);
480        int blue = (int) Math.round(c.getBlue() * 255.0);
481        int alpha = (int) Math.round(c.getOpacity() * 255.0);
482        if (alpha == 255) {
483            return String.format("#%02x%02x%02x", red, green, blue); //NOI18N
484        } else {
485            return String.format("#%02x%02x%02x%02x", red, green, blue, alpha); //NOI18N
486        }
487    }

507        if (property == null) {
508            return bf.toString();
509        }
510        StringBuilder builder = new StringBuilder();
511        if (property.equals("-fx-background-color")) { //NOI18N
512            Paint p = bf.getFill();
513            builder.append(paintToString(p));
514        } else {
515            if (property.equals("-fx-background-insets")) { //NOI18N

517            } else {
518                //the top right, bottom right, bottom left, and top left
519                if (property.equals("-fx-background-radius")) { //NOI18N
520                    handleCornerRadii(bf.getRadii(), builder);
521                }
522            }
523        }
524        return builder.toString();
525    }

536    private static String backgroundImageToString(String property, BackgroundImage bi) {
537        if (property == null) {
538            return bi.toString();
539        }
540        StringBuilder builder = new StringBuilder();
541        if (property.equals("-fx-background-image")) { //NOI18N
542            Image p = bi.getImage();
543            builder.append(Deprecation.getUrl(p));
544        } else {
545            if (property.equals("-fx-background-position")) {             //NOI18N
546                double left = 0, right = 0, top = 0, bottom = 0;
547                if (bi.getPosition().getHorizontalSide() == Side.LEFT) {
548                    left = bi.getPosition().getHorizontalPosition();
549                } else {
550                    right = bi.getPosition().getHorizontalPosition();
551                }
552                if (bi.getPosition().getVerticalSide() == Side.TOP) {
553                    top = bi.getPosition().getVerticalPosition();
554                } else {
555                    bottom = bi.getPosition().getVerticalPosition();
556                }
557                builder.append("left:"); //NOI18N
558                builder.append(EditorUtils.valAsStr(left));
559                builder.append(" right:"); //NOI18N
560                builder.append(EditorUtils.valAsStr(right));
561                builder.append(" top:"); //NOI18N
562                builder.append(EditorUtils.valAsStr(top));
563                builder.append(" bottom:"); //NOI18N
564                builder.append(EditorUtils.valAsStr(bottom));
565            } else {
566                if (property.equals("-fx-background-repeat")) {          //NOI18N
567                    if (bi.getRepeatX() != null) {
568                        builder.append(bi.getRepeatX().toString());
569                    } else {
570                        if (bi.getRepeatY() != null) {
571                            builder.append(bi.getRepeatY().toString());
572                        } else {
573                            builder.append("unknown repeat"); //NOI18N
574                        }
575                    }
576                } else {
577                    if (property.equals("-fx-background-size")) { //NOI18N
578                        BackgroundSize bs = bi.getSize();
579                        if (bs.isContain()) {
580                            builder.append("contain"); //NOI18N
581                        } else {
582                            if (bs.isCover()) {
583                                builder.append("cover"); //NOI18N
584                            } else {
585                                if (bs.getWidth() == BackgroundSize.AUTO) {
586                                    builder.append("width: auto"); //NOI18N
587                                } else {
588                                    builder.append("width: ").append(EditorUtils.valAsStr(bs.getWidth())); //NOI18N
589                                }
590                                if (bs.getHeight() == BackgroundSize.AUTO) {
591                                    builder.append("height: auto"); //NOI18N
592                                } else {
593                                    builder.append("height: ").append(EditorUtils.valAsStr(bs.getHeight())); //NOI18N
594                                }
595                            }
596                        }
597                    }
598                }
599            }
600        }
601        return builder.toString();
602    }

612        } else {
613            if (property.equals("-fx-background-position")) {             //NOI18N
614

662    private static String borderStrokeToString(String property, BorderStroke bs) {
663        if (property == null) {
664            return bs.toString();
665        }
666        StringBuilder builder = new StringBuilder();
667        //top, right, bottom, and left 
668        if (property.equals("-fx-border-color")) { //NOI18N
669            if (bs.getTopStroke().equals(bs.getBottomStroke())
670                    && bs.getRightStroke().equals(bs.getBottomStroke())
671                    && bs.getLeftStroke().equals(bs.getBottomStroke())) {
672                builder.append(paintToString(bs.getBottomStroke()));
673            } else {
674                builder.append(paintToString(bs.getTopStroke())).append(" "); //NOI18N
675                builder.append(paintToString(bs.getRightStroke())).append(" "); //NOI18N
676                builder.append(paintToString(bs.getBottomStroke())).append(" "); //NOI18N
677                builder.append(paintToString(bs.getLeftStroke()));
678            }
679        } else {
680            if (property.equals("-fx-border-insets")) { //NOI18N
681                builder.append(insetsValue(bs.getInsets()));
682            } else {
683                //the top right, bottom right, bottom left, and top left
684                if (property.equals("-fx-border-radius")) { //NOI18N
685                    handleCornerRadii(bs.getRadii(), builder);
686                } else {
687                    if (property.equals("-fx-border-style")) { //NOI18N
688                        builder.append(bs.getTopStyle().toString()).append(", "); //NOI18N
689                        builder.append(bs.getRightStyle().toString()).append(", "); //NOI18N
690                        builder.append(bs.getBottomStyle().toString()).append(", "); //NOI18N
691                        builder.append(bs.getLeftStyle().toString());
692                    } else {
693                        if (property.equals("-fx-border-width")) { //NOI18N
694                            BorderWidths bw = bs.getWidths();
695                            if (MathUtils.equals(bw.getTop(), bw.getBottom())
696                                    && MathUtils.equals(bw.getRight(), bw.getBottom())
697                                    && MathUtils.equals(bw.getLeft(), bw.getBottom())) {
698                                builder.append(EditorUtils.valAsStr(bw.getBottom()));
699                            } else {
700                                builder.append(EditorUtils.valAsStr(bw.getTop())).append(" "). //NOI18N
701                                        append(EditorUtils.valAsStr(bw.getRight())).append(" "). //NOI18N
702                                        append(EditorUtils.valAsStr(bw.getBottom())).append(" "). //NOI18N
703                                        append(EditorUtils.valAsStr(bw.getLeft()));
704                            }
705                        }
706                    }
707                }
708            }
709        }
710        return builder.toString();
711    }

713    private static String paintToString(Paint p) {
714        if (p instanceof Color) {
715            return getColorAsString((Color) p).toLowerCase(Locale.ROOT);
716        } else {
717            String gradient = p.toString();
718            // Workaround for RT-22910
719            gradient = gradient.replaceAll("0x", "#");//NOI18N
720            gradient = removeDotZeroPxPercent(gradient);
721            return gradient;
722        }
723    }

725    private static String fontToString(String property, Font font) {
726        if (property == null) {
727            return removeAllDotZero(font.toString());
728        }
729        StringBuilder builder = new StringBuilder();
730        if (property.equals("-fx-font")) { //NOI18N
731            String size = EditorUtils.valAsStr(font.getSize()); //NOI18N
732            String previewStr = font.getFamily() + " " + size + "px" //NOI18N
733                    + (!font.getName().equals(font.getFamily())
734                    && !"Regular".equals(font.getStyle()) //NOI18N
735                    ? " (" + font.getStyle() + ")" : ""); //NOI18N
736            builder.append(previewStr);
737        } else {
738            if (property.equals("-fx-font-size")) { //NOI18N
739                double p = font.getSize();
740                builder.append(EditorUtils.valAsStr(p)).append("px"); //NOI18N
741
742            } else {
743                if (property.equals("-fx-font-family")) { //NOI18N
744                    builder.append(font.getFamily());
745                } else {
746                    if (property.equals("-fx-font-weight")) { //NOI18N
747                        // There is no such property.
748                        builder.append(removeAllDotZero(font.toString()));
749                    } else {
750                        if (property.equals("-fx-font-style")) { //NOI18N
751                            builder.append(font.getStyle());
752                        }
753                    }
754                }
755            }
756        }
757        return builder.toString();
758    }

772        StringBuilder strBuild = new StringBuilder();
773        Effect adding = effect;
774        while (adding != null) {
775            strBuild.append(adding.getClass().getSimpleName());
776            adding = getEffectInput(adding);
777            if (adding != null) {
778                strBuild.append(", "); //NOI18N

780        }
781        return strBuild.toString();

872    private static String removeDotZeroPxPercent(String str) {
873        // Remove ".0" in strings, for "px" and "%" notations
874        str = str.replaceAll("\\.0px", "px"); //NOI18N
875        str = str.replaceAll("\\.0em", "em"); //NOI18N
876        str = str.replaceAll("\\.0\\%", "%"); //NOI18N
877        return str;
878    }

886    private static void handleCornerRadii(CornerRadii cr, StringBuilder builder) {
887        // Each radius has a vertical and horizontal radius
888        // See  http://www.w3.org/TR/css3-background/#the-border-radius 
889
890        double topLeftH = cr.getTopLeftHorizontalRadius();
891        double topLeftV = cr.getTopLeftVerticalRadius();
892        double topRightH = cr.getTopRightHorizontalRadius();
893        double topRightV = cr.getTopRightVerticalRadius();
894        double bottomLeftH = cr.getBottomLeftHorizontalRadius();
895        double bottomLeftV = cr.getBottomLeftVerticalRadius();
896        double bottomRightH = cr.getBottomRightHorizontalRadius();
897        double bottomRightV = cr.getBottomRightVerticalRadius();
898
899        if (MathUtils.equals(topLeftH, topLeftV) && MathUtils.equals(topRightH, topRightV)
900                && MathUtils.equals(bottomLeftH, bottomLeftV) && MathUtils.equals(bottomRightH, bottomRightV)) {
901            if (MathUtils.equals(topLeftH, topRightH) && MathUtils.equals(topRightH, bottomLeftH)
902                    && MathUtils.equals(bottomLeftH, bottomRightH)) {
903                // Same radius for all => single value
904                builder.append(EditorUtils.valAsStr(topLeftH));
905            } else {
906                // Same value for vertical and horizontal radii 
907                // => 4 values for topLeft, topRight, bottomLeft, bottomRight
908                builder.append(EditorUtils.valAsStr(topLeftH)).append(" "). //NOI18N
909                        append(EditorUtils.valAsStr(topRightH)).append(" "). //NOI18N
910                        append(EditorUtils.valAsStr(bottomRightH)).append(" "). //NOI18N
911                        append(EditorUtils.valAsStr(bottomLeftH));
912            }
913        } else {
914            // Separate value for each.
915            // Syntax: "horizontal values / vertical values"
916            builder.append(EditorUtils.valAsStr(topLeftH)).append(" "). //NOI18N
917                    append(EditorUtils.valAsStr(topRightH)).append(" "). //NOI18N
918                    append(EditorUtils.valAsStr(bottomRightH)).append(" "). //NOI18N
919                    append(EditorUtils.valAsStr(bottomLeftH)).
920                    append(" / ").//NOI18N
921                    append(EditorUtils.valAsStr(topLeftV)).append(" "). //NOI18N
922                    append(EditorUtils.valAsStr(topRightV)).append(" "). //NOI18N
923                    append(EditorUtils.valAsStr(bottomRightV)).append(" "). //NOI18N
924                    append(EditorUtils.valAsStr(bottomLeftV));
925        }
926    }
