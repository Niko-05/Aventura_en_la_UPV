330    /**
331     * Scrolls this content panel so that the selected objects are visible.
332     */
333    public void scrollToSelection() {
334        // Walk through the selected objects and computes the enclosing bounds.
335        final BoundsUnion union = new BoundsUnion();
336        final Selection selection = getEditorController().getSelection();
337        if (selection.getGroup() instanceof ObjectSelectionGroup) {
338            final ObjectSelectionGroup osg = (ObjectSelectionGroup) selection.getGroup();
339            for (FXOMObject i : osg.getItems()) {
340                final DesignHierarchyMask mask = new DesignHierarchyMask(i);
341                final FXOMObject nodeFxomObject = mask.getClosestFxNode();
342                if (nodeFxomObject != null) {
343                    final Node node = (Node) nodeFxomObject.getSceneGraphObject();
344                    union.add(node.localToScene(node.getLayoutBounds()));
345                }
346            }
347        }
348
349        if (union.getResult() != null) {
350            final Node content = scrollPane.getContent();
351            final Bounds enclosing = content.sceneToLocal(union.getResult());
352            final ScrollPaneBooster spb = new ScrollPaneBooster(scrollPane);
353            spb.scrollTo(enclosing);
354        }
355    }

358    public void reveal(FXOMObject targetFxomObject) {
359        FXOMObject fxomObject = targetFxomObject;
360        
361        while (fxomObject != null) {
362            final Object sceneGraphObject = fxomObject.getSceneGraphObject();
363            
364            if (sceneGraphObject instanceof Tab) {
365                final Tab tab = (Tab) sceneGraphObject;
366                final TabPane tabPane = tab.getTabPane();
367                assert tabPane != null;
368                tabPane.getSelectionModel().select(tab);
369            } else if (sceneGraphObject instanceof TitledPane) {
370                final TitledPane titledPane = (TitledPane) sceneGraphObject;
371                if (titledPane.getParent() instanceof Accordion) {
372                    final Accordion accordion = (Accordion) titledPane.getParent();
373                    accordion.setExpandedPane(titledPane);
374                }
375            }
376            
377            DesignHierarchyMask mask = new DesignHierarchyMask(fxomObject);
378            fxomObject = mask.getParentFXOMObject();
379        }
380    }

394    /**
395     * Returns the topmost FXOMObject at (sceneX, sceneY) but ignoring
396     * objects from the exclude set.
397     * 
398     * @param sceneX x coordinate of a scene point
399     * @param sceneY y coordinate of a scene point
400     * @param excludes null or a set of FXOMObject to be excluded from the pick.
401     * @return null or the topmost FXOMObject located at (sceneX, sceneY)
402     */
403    public FXOMObject pick(double sceneX, double sceneY, Set<FXOMObject> excludes) {
404        final FXOMObject result;
405        
406        if (isContentDisplayable()) {
407            final FXOMDocument fxomDocument = getEditorController().getFxomDocument();
408            assert fxomDocument != null;
409            if ((fxomDocument.getFxomRoot() == null) 
410                    || excludes.contains(fxomDocument.getFxomRoot())) {
411                result = null;
412            } else {
413                assert fxomDocument.getFxomRoot().getSceneGraphObject() instanceof Node;
414                result = pick(fxomDocument.getFxomRoot(), sceneX, sceneY, excludes);
415            }
416        } else {
417            result = null;
418        }
419        
420        return result;
421    }

424    /**
425     * Returns the topmost FXOMObject at (sceneX, sceneY) but ignoring
426     * objects from the exclude set and starting the search from startObject.
427     * 
428     * @param startObject starting point of the search
429     * @param sceneX x coordinate of a scene point
430     * @param sceneY y coordinate of a scene point
431     * @param excludes null or a set of FXOMObject to be excluded from the pick.
432     * @return null or the topmost FXOMObject located at (sceneX, sceneY)
433     */
434    public FXOMObject pick(
435            FXOMObject startObject, 
436            double sceneX, 
437            double sceneY,
438            Set<FXOMObject> excludes) {
439        
440        final FXOMObject result;
441        
442        assert isContentDisplayable();
443        assert startObject != null;
444        assert startObject.getSceneGraphObject() instanceof Node;
445        assert excludes != null;
446        assert excludes.contains(startObject) == false;
447        
448        picker.getExcludes().clear();
449        for (FXOMObject exclude : excludes) {
450            if (exclude.getSceneGraphObject() instanceof Node) {
451                picker.getExcludes().add((Node) exclude.getSceneGraphObject());
452            }
453        }
454
455        final Node startNode = (Node) startObject.getSceneGraphObject();
456        final List<Node> hitNodes = picker.pick(startNode, sceneX, sceneY);
457        if (hitNodes == null) {
458            result = null;
459        } else {
460            assert hitNodes.isEmpty() == false;
461
462            FXOMObject hitObject = null;
463            final Iterator<Node> it = hitNodes.iterator();
464            while ((hitObject == null) && it.hasNext()) {
465                final Node hitNode = it.next();
466                hitObject = searchWithNode(hitNode, sceneX, sceneY);
467                if (excludes.contains(hitObject)) {
468                    hitObject = null;
469                }
470            }
471            result = hitObject;
472        }
473        
474        return result;
475    }

477    /**
478     * Returns the FXOMObject which matches (sceneGraphNode, sceneX, sceneY).
479     * 
480     * @param sceneGraphNode scene graph node
481     * @param sceneX x coordinate of a scene point
482     * @param sceneY y coordinate of a scene point
483     * @return an FXOMObject that matches (sceneGraphNode, sceneX, sceneY)
484     */
485    public FXOMObject searchWithNode(Node sceneGraphNode, double sceneX, double sceneY) {
486       final FXOMObject result;
487        
488        final FXOMDocument fxomDocument 
489                = getEditorController().getFxomDocument();
490        final FXOMObject match
491                = fxomDocument.searchWithSceneGraphObject(sceneGraphNode);
492         /*
493         * Refine the search.
494         * With the logic above, a click in a 'tab header' returns the
495         * fxom object associated to the 'tab pane'. We would like to get
496         * the fxom object associated to the 'tab'. When the pick result is
497         * a 'TabPane' we need to refine this result. This refinement logic
498         * is available in AbstractDriver.
499         */
500        if (match != null) {
501            final AbstractDriver driver = lookupDriver(match);
502            result = driver.refinePick(sceneGraphNode, sceneX, sceneY, match);
503        } else {
504            result = null;
505        }
506        
507        return result;
508    }

591    /**
592     * Computes the transform that projects from local coordinates of a 
593     * scene graph object to the rudder layer local coordinates.
594     * @param sceneGraphObject a scene graph object
595     * @return transform from sceneGraphObject local coordinates to rudder local coordinates
596     */
597    public Transform computeSceneGraphToRudderLayerTransform(Node sceneGraphObject) {
598        assert sceneGraphObject != null;
599        assert sceneGraphObject.getScene() == rudderLayer.getScene();
600        
601        final Transform t1 = sceneGraphObject.getLocalToSceneTransform();
602        final Transform t2 = rudderLayer.getLocalToSceneTransform();
603        final Transform result;
604        
605        try {
606            final Transform i2 = t2.createInverse();
607            result = i2.createConcatenation(t1);
608        } catch(NonInvertibleTransformException x) {
609            throw new RuntimeException(x);
610        }
611        
612        return result;
613    }

651    /**
652     * @treatAsPrivate
653     * Returns true if this content panel is able to display the content ie
654     * 1) fxomDocument != null
655     * 2) (fxomDocument.getFxomRoot() == null) or fxomDocument.getFxomRoot().isNode()
656     * 3) workspaceController.getLayoutException() == null
657     * 
658     * @return true if this content panel is able to display the content
659     */
660    public boolean isContentDisplayable() {
661        final boolean result;
662        
663        final FXOMDocument fxomDocument = getEditorController().getFxomDocument();
664        if (fxomDocument == null) {
665            result = false;
666        } else if (fxomDocument.getFxomRoot() == null) {
667            result = true;
668        } else {
669            result = fxomDocument.getFxomRoot().isNode()
670                    && workspaceController.getLayoutException() == null;
671        }
672        
673        return result;
674    }

680    /**
681     * @treatAsPrivate fxom document has changed
682     * @param oldDocument old fxom document
683     */
684    @Override
685    protected void fxomDocumentDidChange(FXOMDocument oldDocument) {
686        if (oldDocument != null) {
687            assert oldDocument.getSceneGraphHolder() == this;
688            oldDocument.endHoldingSceneGraph();
689        }
690        final FXOMDocument fxomDocument = getEditorController().getFxomDocument();
691        if (fxomDocument != null) {
692            assert fxomDocument.getSceneGraphHolder() == null;
693            fxomDocument.beginHoldingSceneGraph(this);
694        }
695        
696        final Exception currentLayoutException
697                = workspaceController.getLayoutException();
698        workspaceController.setFxomDocument(fxomDocument);
699        final Exception newLayoutException
700                = workspaceController.getLayoutException();
701        if ((newLayoutException != null) && (newLayoutException != currentLayoutException)) {
702            getEditorController().getMessageLog().logWarningMessage(
703                    "log.warning.layout.failed", newLayoutException.getMessage());
704        }
705        
706        if (isOutlinesVisible()) {
707            updateOutlines();
708        }
709        if (currentModeController != null) {
710            currentModeController.fxomDocumentDidChange(oldDocument);
711        }
712
713        resetViewport();
714    }

884     * @param fxomObject an fxom object (never null)

887    public AbstractDriver lookupDriver(FXOMObject fxomObject) {
888        final Object sceneGraphObject = fxomObject.getSceneGraphObject();
889        final AbstractDriver result;
890        
891        if (sceneGraphObject instanceof HBox) {
892            result = new HBoxDriver(this);
893        } else if (sceneGraphObject instanceof VBox) {
894            result = new VBoxDriver(this);
895        } else if (sceneGraphObject instanceof GridPane) {
896            result = new GridPaneDriver(this);
897        } else if (sceneGraphObject instanceof BorderPane) {
898            result = new BorderPaneDriver(this);
899        } else if (sceneGraphObject instanceof Line) {
900            result = new LineDriver(this);
901        } else if (sceneGraphObject instanceof FlowPane) {
902            result = new FlowPaneDriver(this);
903        } else if (sceneGraphObject instanceof TextFlow) {
904            result = new TextFlowDriver(this);
905        } else if (sceneGraphObject instanceof ToolBar) {
906            result = new ToolBarDriver(this);
907        } else if (sceneGraphObject instanceof SplitPane) {
908            result = new SplitPaneDriver(this);
909        } else if (sceneGraphObject instanceof Tab) {
910            result = new TabDriver(this);
911        } else if (sceneGraphObject instanceof TabPane) {
912            result = new TabPaneDriver(this);
913        } else if (sceneGraphObject instanceof TableView) {
914            result = new TableViewDriver(this);
915        } else if (sceneGraphObject instanceof TableColumn) {
916            result = new TableColumnDriver(this);
917        } else if (sceneGraphObject instanceof TreeTableView) {
918            result = new TreeTableViewDriver(this);
919        } else if (sceneGraphObject instanceof TreeTableColumn) {
920            result = new TreeTableColumnDriver(this);
921        } else if (sceneGraphObject instanceof Node) {
922            result = new GenericDriver(this);
923        } else {
924            result = null;
925        }
926        
927        return result;
928    }

1015    private void installStylingIsolationGroup() {
1016        assert contentGroup.getParent() == scalingGroup;
1017        
1018        /*
1019         * To isolate user content styling from SB's own styling we 
1020         * insert two custom groups between scalingGroup and contentGroup:
1021         * 
1022         * 1) original layout loaded from FXML:
1023         * 
1024         *      ...
1025         *          scalingGroup
1026         *              contentGroup
1027         *                  ... (user content)
1028         * 
1029         * 2) layout with isolation groups:
1030         * 
1031         *      ... 
1032         *          scalingGroup
1033         *              isolationGroupA     # impl_getAllParentStylesheets() overriden
1034         *                  isolationGroupB     # getStyleableParent() overriden 
1035         *                      contentGroup
1036         *                          ... (user content)
1037         *
1038         * isolationGroupA prevents styling from SB to apply to user content
1039         * isolationGroupB enables user content to have its own theme (modena, caspian...)
1040         */
1041        
1042        final Group isolationGroupA = Deprecation.makeStylingIsolationGroupA();
1043        final Group isolationGroupB = Deprecation.makeStylingIsolationGroupB();
1044
1045        final int contentGroupIndex = scalingGroup.getChildren().indexOf(contentGroup);
1046        assert contentGroupIndex != -1;
1047        scalingGroup.getChildren().remove(contentGroup);
1048        scalingGroup.getChildren().add(contentGroupIndex, isolationGroupA);
1049        isolationGroupA.getChildren().add(isolationGroupB);
1050        isolationGroupB.getChildren().add(contentGroup);
1051    }

1077    private void updateOutlines() {
1078        assert isOutlinesVisible();
1079        
1080        // Collects fxom objects associated to a node in the fxom document
1081        final List<FXOMObject> allNodes = collectNodes();
1082        
1083        for (int i = 0, count = allNodes.size(); i < count; i++) {
1084            assert allNodes.get(i) instanceof FXOMInstance;
1085            final FXOMInstance nodeInstance = (FXOMInstance) allNodes.get(i);
1086            if (i < outlines.size()) {
1087                final NodeOutline currentOutline = outlines.get(i);
1088                if (currentOutline.getFxomObject() != nodeInstance) {
1089                    replaceOutline(i, nodeInstance);
1090               } else {
1091                    switch(currentOutline.getState()) {
1092                        case CLEAN:
1093                            break;
1094                        case NEEDS_RECONCILE:
1095                            // scene graph associated to currentOutline has changed but h is still compatible
1096                            currentOutline.reconcile();
1097                            break;
1098                        case NEEDS_REPLACE:
1099                            // currentOutline is no longer compatible with the new scene graph object 
1100                            replaceOutline(i, nodeInstance);
1101                            break;
1102                    }
1103                }
1104            } else {
1105                addOutline(outlines.size(), nodeInstance);
1106            }
1107        }
1108        for (int i = allNodes.size(), count = outlines.size(); i < count; i++) {
1109            removeOutline(allNodes.size());
1110        }
1111        assert outlines.size() == allNodes.size();
1112    }

1114    private void addOutline(int i, FXOMInstance nodeInstance) {
1115        assert outlines.size() == outlineLayer.getChildren().size();
1116        
1117        final NodeOutline newOutline = new NodeOutline(this, nodeInstance);
1118        outlines.add(i, newOutline);
1119        outlineLayer.getChildren().add(i, newOutline.getRootNode());
1120        
1121        assert outlines.size() == outlineLayer.getChildren().size();
1122        assert outlines.get(i).getRootNode() == outlineLayer.getChildren().get(i);
1123    }

1125    private void replaceOutline(int i, FXOMInstance nodeInstance) {
1126        removeOutline(i);

1141    private List<FXOMObject> collectNodes() {
1142        final List<FXOMObject> result = new ArrayList<>();
1143        
1144        final List<FXOMObject> candidates = new ArrayList<>();
1145        final FXOMDocument fxomDocument = getEditorController().getFxomDocument();
1146        if ((fxomDocument != null) && (fxomDocument.getFxomRoot() != null)) {
1147            candidates.add(fxomDocument.getFxomRoot());
1148        } 
1149        
1150        while (candidates.isEmpty() == false) {
1151            final FXOMObject candidate = candidates.get(0);
1152            candidates.remove(0);
1153            if (candidate.isNode()) {
1154                final Node sgo = (Node) candidate.getSceneGraphObject();
1155                if (sgo.getScene() == getPanelRoot().getScene()) {
1156                    result.add(candidate);
1157                }
1158            }
1159            final DesignHierarchyMask m = new DesignHierarchyMask(candidate);
1160            if (m.isAcceptingSubComponent()) {
1161                for (int i = 0, c = m.getSubComponentCount(); i < c; i++) {
1162                    final FXOMObject subComponent = m.getSubComponentAtIndex(i);
1163                    candidates.add(subComponent);
1164                }
1165            }
1166            for (DesignHierarchyMask.Accessory a : DesignHierarchyMask.Accessory.values()) {
1167                if (m.isAcceptingAccessory(a)) {
1168                    final FXOMObject accessoryObject = m.getAccessory(a);
1169                    if ((accessoryObject != null) && accessoryObject.isNode()) {
1170                        candidates.add(accessoryObject);
1171                    }
1172                }
1173            }
1174        }
1175        
1176        return result;
