307        public List<CssContentMaker.CssPropertyState.CssStyle> getFxThemeHiddenByModel() {
308            List<CssContentMaker.CssPropertyState.CssStyle> ret = new ArrayList<>();
309            CssContentMaker.CssPropertyState ps = getWinner();
310            List<CssContentMaker.CssPropertyState.CssStyle> notAppliedStyles = 
311                    ps == null ? 
312                    Collections.<CssContentMaker.CssPropertyState.CssStyle>emptyList() : 
313                    ps.getNotAppliedStyles();
314            boolean hasModel = modelState().get() != null;
315            if (hasModel) {
316                List<Style> allStyles = Deprecation.getMatchingStyles(getStyleable(), target);
317                List<Style> matchingStyles = CssContentMaker.removeUserAgentStyles(allStyles);
318                for (Style style : matchingStyles) {
319                    CssContentMaker.CssPropertyState.CssStyle cssStyle = new CssContentMaker.CssPropertyState.CssStyle(style);
320                    if (cssStyle.getOrigin() == StyleOrigin.USER_AGENT && !notAppliedStyles.contains(cssStyle)) {
321                        if (getStyleable().getProperty().equals(cssStyle.getCssProperty())) {
322                            cssStyle = CssContentMaker.retrieveStyle(matchingStyles, style);
323                            ret.add(cssStyle);
324                        }
325                    }
326                }
327            }
328            return ret;
329        }

333    private CssContentMaker.CssPropertyState retrieveCssStyle(
334            CssMetaData<?, ?> cssMeta, Collection<CssContentMaker.CssPropertyState> styles) {
335        for (CssContentMaker.CssPropertyState prop : styles) {
336            if (prop.getCssProperty().equals(cssMeta.getProperty())) {
337                return prop;
338            } else {
339                if (prop.getSubProperties() != null) {
340                    for (CssContentMaker.PropertyState sub : prop.getSubProperties()) {
341                        if (sub.getCssProperty().equals(cssMeta.getProperty())) {
342                            return (CssContentMaker.CssPropertyState) sub;
343                        }
344                    }
345                }
346            }
347        }
348        return null;
349    }

355    @SuppressWarnings("rawtypes")
356    public final Collection<CssProperty> getAllStyleables() {
357        if (props == null) {
358            props = new TreeSet<>();
359            List<CssMetaData<? extends Styleable, ?>> cssMetaList = node.getCssMetaData();
360            for (CssMetaData<? extends Styleable, ?> cssMeta : cssMetaList) {
361                CssProperty mainProp = new CssProperty(this, cssMeta, node, fxomObject);
362                props.add(mainProp);
363                if (cssMeta.getSubProperties() != null) {
364                    for (CssMetaData sub : cssMeta.getSubProperties()) {
365                        CssProperty subProp = new CssProperty(this, mainProp, sub, node, fxomObject);
366                        mainProp.getSubProperties().add(subProp);
367                    }
368                }
369            }
370
371            if (node instanceof Skinnable) {
372                Skinnable skinnable = (Skinnable) node;
373                Node skinNode = skinnable.getSkin().getNode();
374                List<CssMetaData<? extends Styleable, ?>> skinList = skinNode.getCssMetaData();
375                for (CssMetaData<? extends Styleable, ?> skinCssMeta : skinList) {
376                    boolean found = false;
377                    for (CssMetaData cssMeta : cssMetaList) {
378                        if (skinCssMeta.getProperty().equals(cssMeta.getProperty())) {
379                            found = true;
380                            break;
381                        }
382                    }
383                    if (!found) {
384                        CssProperty mainProp = new CssProperty(this, skinCssMeta, skinNode, fxomObject);
385                        props.add(mainProp);
386                        if (skinCssMeta.getSubProperties() != null) {
387                            for (CssMetaData<? extends Styleable, ?> sub : skinCssMeta.getSubProperties()) {
388                                CssProperty subProp = new CssProperty(this, sub, node, fxomObject);
389                                mainProp.getSubProperties().add(subProp);
390                            }
391                        }
392                    }
393                }
394            }
395        }
396        return props;
397    }

420    public static class RuleComparator implements Comparator<MatchingRule> {
421
422        @Override
423        public int compare(MatchingRule t, MatchingRule t1) {
424            int originComparaison = compareOrigin(
425                    CssInternal.getOrigin(t.getRule()), CssInternal.getOrigin(t1.rule));
426            int tnotApplied = countNotApplied(t.declarations);
427            int t1notApplied = countNotApplied(t1.declarations);
428            int notAppliedComparaisons = tnotApplied - t1notApplied;
429
430            if (originComparaison == 0) {// Same origin, not Applied count is what is important. The less, the stronger
431                return notAppliedComparaisons;
432            } else {
433                return originComparaison;
434            }
435        }
436
437    }

439    private static int compareOrigin(StyleOrigin toCompare, StyleOrigin other) {
440        int index1 = ORDERED_ORIGIN.indexOf(toCompare);
441        int index2 = ORDERED_ORIGIN.indexOf(other);
442        return index2 - index1;
443
444    }

446    private static int countNotApplied(List<MatchingDeclaration> declarations) {
447        int count = 0;
448        for (MatchingDeclaration decl : declarations) {
449            if (!decl.isApplied()) {
450                count += 1;
451            }
452        }
453        return count;
454    }

548            Collection<CssContentMaker.CssPropertyState> source, 
549            Collection<CssContentMaker.CssPropertyState> target) {
550        for (CssContentMaker.CssPropertyState p : source) {
551            if (p.getSubProperties().isEmpty()) {
552                target.add(p);
553            } else {
554                for (CssContentMaker.PropertyState sub : p.getSubProperties()) {

556                }
557            }

561    // Sorted according to Author/User Agent and applied / not applied
562    public final List<MatchingRule> getMatchingRules() {
563        if (matchingRules == null) {
564            Collection<CssContentMaker.CssPropertyState> styledProperties = new TreeSet<>();
565            addSubProperties(getUserAgentStyles(), styledProperties);
566            addSubProperties(getAuthorStyles(), styledProperties);
567                // We need them to have the exhaustive set of styled properties.
568            // We compute the rules based on the set of properties.
569            addSubProperties(getInlineStyles(), styledProperties);
570            matchingRules = new HashMap<>();
571            for (CssContentMaker.CssPropertyState cssP : styledProperties) {
572                List<CssContentMaker.PropertyState> l = cssP.getSubProperties();
573                if (l.isEmpty()) {
574                    addMatchingDeclaration(cssP);
575                } else {
576                    for (CssContentMaker.PropertyState pp : l) {
577                        CssContentMaker.CssPropertyState cssSubP = (CssContentMaker.CssPropertyState) pp;
578                        addMatchingDeclaration(cssSubP);
579                    }
580                }
581            }
582            for (Map.Entry<MatchingRule, List<MatchingDeclaration>> entry : matchingRules.entrySet()) {
583                MatchingRule rule = entry.getKey();
584                // Filterout the Inline
585                if (CssInternal.getOrigin(rule.getRule()) != StyleOrigin.INLINE) {
586                    rule.addDeclarations(entry.getValue());
587                    sortedMatchingRules.add(rule);
588                }
589            }
590            Collections.<MatchingRule>sort(sortedMatchingRules, new RuleComparator());
591        }
592        return sortedMatchingRules;
593    }

595    private void addMatchingDeclaration(CssPropertyState cssP) {
596        addMatchingDeclaration(cssP, cssP.getStyle(), true, false);
597        for (CssPropertyState.CssStyle s : cssP.getNotAppliedStyles()) {
598            addMatchingDeclaration(cssP, s, false, false);
599        }
600    }
601
602    private void addMatchingDeclaration(
603            CssPropertyState cssP, CssPropertyState.CssStyle style, boolean applied, boolean isLookup) {
604        MatchingRule mr = new MatchingRule(style.getCssRule(), style.getSelector());
605        List<MatchingDeclaration> lst = matchingRules.get(mr);
606        if (lst == null) {
607            lst = new ArrayList<>();
608            matchingRules.put(mr, lst);
609        }
610        MatchingDeclaration pmr = new MatchingDeclaration(style, cssP, applied, isLookup);
611        boolean found = false;
612        for (MatchingDeclaration d : lst) {
613            if (d.style.getCssProperty().equals(style.getCssProperty())) {
614                found = true;
615                break;
616            }
617        }
618
619        if (!found) {
620            lst.add(pmr);
621        }
622        for (CssContentMaker.CssPropertyState.CssStyle lookup : style.getLookupChain()) {
623            addMatchingDeclaration(cssP, lookup, applied, true);
624        }
625    }

669                 * and background-color set by inline or author.
670                 */
671                if (cssMetaList.getSubProperties() != null) {
672                    for (CssMetaData sub : cssMetaList.getSubProperties()) {
673                        List<CssContentMaker.CssPropertyState.CssStyle> notApplied = CssContentMaker.getNotAppliedStyles(entry.getValue(), node, sub);
674                        for (Style style : entry.getValue()) {
675                            StyleOrigin styleOrigin = CssInternal.getOrigin(style);
676                            if (style.getDeclaration().getProperty().equals(sub.getProperty())
677                                    && (styleOrigin == origin)) {
678                                CssContentMaker.CssPropertyState.CssStyle cssStyle = CssContentMaker.retrieveStyle(entry.getValue(), style);
679                                String subCssValue = CssValueConverter.toCssString(sub.getProperty(), style.getDeclaration().getRule(), value.getValue());
680                                CssContentMaker.CssSubPropertyState subCss = new CssContentMaker.CssSubPropertyState(value, sub, subCssValue);
681                                subCss.setStyle(cssStyle);
682                                subCss.getNotAppliedStyles().addAll(notApplied);
683                                pState.getSubProperties().add(subCss);
684                            }
685                        }
686                    }
687                    // eg: -fx-font set
688                    CssContentMaker.CssPropertyState.CssStyle style = CssContentMaker.retrieveStyle(entry.getValue(), st);
689                    pState.setStyle(style);
690                    if (!st.getDeclaration().getProperty().equals(cssMetaList.getProperty())) {
691                        style.setUnused();
692                    }
693                } else {
694                        // Single style for this single property.
695                    // Transform the flat list into a chain of lookup.
696                    CssContentMaker.CssPropertyState.CssStyle style = CssContentMaker.retrieveStyle(entry.getValue(), st);
697                    pState.setStyle(style);
698                }
699                List<Style> applied = new ArrayList<>();
700                applied.add(st);
701                pState.getNotAppliedStyles().addAll(CssContentMaker.getNotAppliedStyles(applied, node, cssMetaList));
702                /*
703                 * In case the origin is not the same and no sub properties have been found for 
704                 * the passed origin, then the property is not taken into consideration
705                 */
706                if (o == origin || !pState.getSubProperties().isEmpty()) {
707                    propertyStates.add(pState);
708                }
709            }
710        }
711        return propertyStates;
712    }
