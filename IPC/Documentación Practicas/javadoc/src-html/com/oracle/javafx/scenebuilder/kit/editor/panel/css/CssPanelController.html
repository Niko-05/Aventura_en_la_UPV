475    public void filter(String pattern) {
476        if (model == null) {
477            return;
478        }
479        ObservableList<CssProperty> filtered;
480        if (pattern == null || pattern.trim().length() == 0) {
481            filtered = model;
482        } else {
483            filtered = FXCollections.observableArrayList();
484            for (CssProperty p : model) {
485                if (p.propertyName().get().contains(pattern.trim())) {
486                    filtered.add(p);
487                }
488            }
489        }
490        updateTable(filtered);
491    }

585    private void updateSelectedObject() {
586        selection = editorController.getSelection();
587        if (!isMultipleSelection()) {
588            if (isPickMode()) {
589                // In pick mode:
590                // If the selected node is the "root" node ==> we get its FXOMInstance
591                // Else, we don't have an FXOMInstance
592                Object pickObject = selection.getCheckedHitNode();
593                FXOMInstance fxomInstance = getFXOMInstance(selection);
594                if (fxomInstance != null && fxomInstance.getSceneGraphObject() == pickObject) {
595                    selectedObject = fxomInstance;
596                } else {
597                    selectedObject = pickObject;
598                }
599//                System.out.println("(pick mode) selectedObject = " + selectedObject);
600            } else {
601                selectedObject = getFXOMInstance(selection);
602//                System.out.println("selectedObject = " + selectedObject);
603            }
604        }
605    }

628    private void setCSSContent() {
629        if (selectedObject instanceof Skinnable) {
630            if (((Skinnable) selectedObject).getSkin() == null) {
631                return;
632            }
633        }
634
635        clearContent();
636        if (isMultipleSelection()) {
637            viewMessage(I18N.getString("csspanel.multiselection"));
638            return;
639        }
640
641        //
642        // Workaround for:
643        //     RT-35742: GridPane content is not updated as it should
644        //
645        // The TableView setItems is put in a runLater() so that 
646        // the TableView is rendered empty first, then with the new content.
647        // This is an ugly (and temporary) workaround, since one can see the TableView flashing on model update.
648        //
649        Platform.runLater(new Runnable() {
650
651            @Override
652            public void run() {
653                if (selectedObject != null) { // Update content.
654                    fillSelectionContent();
655                    collectCss();
656                }
657            }
658        });
659    }

716        Node enclosingNode = getEnclosingNode(fxomDoc, node);
717        // The componentRootNode can be a skin structure (Tab, Column), in this case the enclosingNode 
718        // is not == to the componentRootNode. That is why we need to compare the enclosingNode of both
719        // n and componentRootNode nodes.
720        Node componentRootNodeEnclosingNode = getEnclosingNode(fxomDoc, componentRootNode);
721        // this is a skin's node and not a node from a component located inside 
722        // the skin (eg: SplitPane content being a Button is not part of the SplitPane Skin.
723        boolean isOtherComponentNode = enclosingNode != componentRootNodeEnclosingNode;
724        if (componentRootNode != node && !node.getStyleClass().isEmpty() && !isOtherComponentNode && !(node instanceof Skin)) {
725            Item ni = new Item(node, createItemName(node), createOptional(node));//NOI18N
726            parentItem.getChildren().add(ni);
727            parentItem = ni;
728        }
729        if (node instanceof Parent && !isOtherComponentNode) {
730            Parent parentNode = (Parent) node;
731            for (Node child : parentNode.getChildrenUnmodifiable()) {
732                addSubStructure(componentRootNode, parentItem, child);
733            }
734        }
735    }

737    private Node getEnclosingNode(FXOMDocument fxomDoc, Node n) {
738        Node node = n;
739        FXOMObject enclosingFXOMObj = fxomDoc.searchWithSceneGraphObject(node);
740        while (enclosingFXOMObj == null) {
741            node = node.getParent();
742            if (node == null) {
743                return null;
744            }
745            enclosingFXOMObj = fxomDoc.searchWithSceneGraphObject(node);
746        }
747        Object enclosingObj = enclosingFXOMObj.getSceneGraphObject();
748        assert enclosingObj instanceof Node;
749        return (Node) enclosingObj;
750    }

839    private void initializeRulesTextPanes(NodeCssState state) {
840        rulesBox.getChildren().clear();
841        List<NodeCssState.MatchingRule> rulesList = state.getMatchingRules();
842        HtmlStyler htmlStyler = new HtmlStyler();
843        rulesTree = new TreeView<>();
844        CopyHandler.attachContextMenu(rulesTree);
845        rulesTree.setShowRoot(false);
846        TreeItem<Node> ruleRoot = new TreeItem<>(new Text(""));//NOI18N
847        rulesTree.setRoot(ruleRoot);
848        for (NodeCssState.MatchingRule rule : rulesList) {
849            List<NodeCssState.MatchingDeclaration> lst = rule.getDeclarations();
850
851            String selector = rule.getSelector();
852
853            String txt = selector + " { ";//NOI18N
854            String source = nonNull(getSource(rule.getRule()));
855            Text text = CopyHandler.makeCopyableNode(new Text(txt + source), txt);
856            TreeItem<Node> start = new TreeItem<>(text);
857            ruleRoot.getChildren().add(start);
858            htmlStyler.cssRuleStart(selector, source);
859
860            for (NodeCssState.MatchingDeclaration p : lst) {
861                CssPropertyState prop = p.getProp();
862                attachStyleProperty(ruleRoot, prop, p.getStyle(), p.isApplied(), p.isLookup());
863                CssStyle style = p.getStyle();
864                String cssValue = CssValueConverter.toCssString(style.getCssProperty(), style.getCssRule(), style.getParsedValue());
865                htmlStyler.addProperty(p.getStyle().getCssProperty(), cssValue, p.isApplied());
866            }
867            TreeItem<Node> end = new TreeItem<>(CopyHandler.createCopyableText("}"));//NOI18N
868            ruleRoot.getChildren().add(end);
869            setTreeHeight(rulesTree);
870            htmlStyler.cssRuleEnd();
871        }
872        if (ruleRoot.getChildren().isEmpty()) {
873            rulesBox.getChildren().add(new Label(NO_MATCHING_RULES));
874        } else {
875            rulesBox.getChildren().add(rulesTree);
876        }
877        if (htmlStyler.isEmpty()) {
878            htmlStyler.addMessage(NO_MATCHING_RULES);
879        }
880        textPane.getEngine().loadContent(htmlStyler.getHtmlString());
881    }
