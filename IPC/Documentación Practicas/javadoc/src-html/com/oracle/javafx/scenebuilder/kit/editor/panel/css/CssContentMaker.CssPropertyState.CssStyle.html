072    public static <N extends Node> PropertyState initialValue(N n, CssMetaData<N, ?> sub) {
073        PropertyState val = null;
074
075        try {
076            Object fxValue;
077            String cssValue;
078            Object value = sub.getInitialValue(n);
079            if (value == null) {
080                cssValue = "none";//NOI18N
081                fxValue = cssValue;
082            } else {
083                fxValue = sub.getInitialValue(n);
084                cssValue = CssValueConverter.toCssString(sub.getProperty(), n);
085            }
086            val = newInitialPropertyState(fxValue, cssValue, n, sub);
087        } catch (RuntimeException ex) {
088            System.out.println(ex.getMessage() + " " + ex);
089            // Ok no initial value or InitialValue bug.
090        }
091        return val;
092    }

118    @SuppressWarnings("rawtypes")
119    public static <N extends Node> PropertyState modelValue(N node, CssMetaData<?, ?> cssMeta, FXOMObject fxomObject) {
120        PropertyState val = null;
121
122        if (fxomObject == null) {
123            // In this case, we are handling a sub-component, no model value then.
124            return null;
125        }
126        // First retrieve the java bean property and check if it is overriden by the inspector.
127        String beanPropName = CssUtils.getBeanPropertyName(node, cssMeta);
128        if (beanPropName == null) {
129            // No corresponding java bean property
130            return null;
131        }
132        PropertyName beanPropertyName = new PropertyName(beanPropName);
133        assert fxomObject instanceof FXOMInstance;
134        FXOMInstance fxomInstance = (FXOMInstance) fxomObject;
135        ValuePropertyMetadata propMeta
136                = Metadata.getMetadata().queryValueProperty(fxomInstance, beanPropertyName);
137        if (propMeta == null) {
138            // No corresponding metadata
139            return null;
140        }
141        if (!propMeta.isReadWrite()) {
142            // R/O : no overridden
143            return null;
144        }
145        boolean overriden = false;
146        Object defaultValue = propMeta.getDefaultValueObject();
147        Object propertyValue = propMeta.getValueObject(fxomInstance);
148        if ((propertyValue == null) || (defaultValue == null)) {
149            if (propertyValue != defaultValue) {
150                overriden = true;
151            }
152        } else if (!propertyValue.equals(defaultValue)) {
153            overriden = true;
154        }
155
156        if (overriden) {
157            // We have an override.
158            val = new BeanPropertyState(propMeta, cssMeta.getProperty(), propertyValue,
159                    CssValueConverter.toCssString(cssMeta.getProperty(), propertyValue));
160            // An overriden can have sub properties
161            if (cssMeta.getSubProperties() != null && !cssMeta.getSubProperties().isEmpty()) {
162                for (CssMetaData sub : cssMeta.getSubProperties()) {
163                    // Create a virtual sub property
164                    PropertyState subProp = new BeanPropertyState(propMeta, sub.getProperty(),
165                            propertyValue, CssValueConverter.toCssString(sub.getProperty(), propertyValue));
166                    val.getSubProperties().add(subProp);
167                }
168            }
169        }
170        return val;
171    }

173    public static Node getSourceNodeForStyle(Object component, String property) {
174        Node ret = null;
175        Node n = CssUtils.getNode(component);
176        if (n != null) {
177            if (n.getStyle() != null && n.getStyle().contains(property)) {
178                ret = n;
179            } else {
180                Parent p = n.getParent();
181                while (p != null) {
182                    String s = p.getStyle();
183                    if (s != null && s.contains(property)) {
184                        ret = p;
185                        break;
186                    }
187                    p = p.getParent();
188                }
189            }
190        }
191        return ret;
192    }

194    public static boolean isInlineInherited(Object component, CssPropertyState cssProperty) {
195        boolean isInherited = false;
196        Node node = CssUtils.getNode(component);
197
198        if (node == null) {
199            return false;
200        }
201
202        // Not located on this node, must be inherited then
203        if (node.getStyle() == null) {
204            return true;
205        }
206
207        if (!containsInStyle(cssProperty, node.getStyle())) {
208            isInherited = true;
209        }
210
211        return isInherited;
212    }

228    public static NodeCssState getCssState(Object selectedObject) {
229        Node node = CssUtils.getSelectedNode(selectedObject);
230        if (node == null) {
231            return null;
232        }
233        Parent p = null;
234        double current = 1;
235        try {
236            if (node.getScene() == null) {
237                // The node is not visible (ContextMenu, Tooltip, ...)
238                // A node MUST be in the scene to allow for CSS content collect,
239                // so we add it (temporarily) to the scene. 
240                Node inScene = CssUtils.getFirstAncestorWithNonNullScene(node);
241                if (inScene == null) {
242                    // May happen if the Content Panel is not present
243                    return null;
244                }
245                p = inScene.getParent();
246                current = node.getOpacity();
247                node.setOpacity(0);
248                CssUtils.addToParent(p, node);
249            }
250            NodeCssState state = new NodeCssState(CssInternal.collectCssState(node), node, getFXOMObject(selectedObject));
251            return state;
252        } finally {
253            if (p != null) {
254                CssUtils.removeFromParent(p, node);
255                node.setOpacity(current);
256            }
257        }
258    }

268    @SuppressWarnings("rawtypes")
269    private static <N extends Node> InitialPropertyState newInitialPropertyState(
270            Object fxValue, String cssValue, N n, CssMetaData<?, ?> cssMeta) {
271        InitialPropertyState val
272                = new InitialPropertyState(cssMeta.getProperty(), fxValue, cssValue);
273        if (cssMeta.getSubProperties() != null && !cssMeta.getSubProperties().isEmpty()) {
274            for (CssMetaData sub : cssMeta.getSubProperties()) {
275                Object subValue
276                        = CssValueConverter.getSubPropertyValue(sub.getProperty(), fxValue);
277                String subCssValue = CssValueConverter.toCssString(subValue);
278                PropertyState subProp
279                        = new InitialPropertyState(sub.getProperty(), subValue, subCssValue);
280                val.getSubProperties().add(subProp);
281            }
282        }
283        return val;
284    }

286    // Retrieve the styles associated to the value. This is the case of lookup (or variable)
287    @SuppressWarnings("rawtypes")
288    protected static CssStyle retrieveStyle(List<Style> styles, Style style) {
289        CssStyle st = new CssStyle(style);
290        ParsedValue parsedValue = style.getDeclaration().getParsedValue();
291        assert parsedValue instanceof ParsedValueImpl;
292        ParsedValueImpl parsedValueImpl = (ParsedValueImpl) parsedValue;
293        if (parsedValueImpl.isContainsLookups() || parsedValueImpl.isLookup()) {
294            retrieveStylesFromParsedValue(styles, st, style.getDeclaration().getParsedValue());
295        }
296        return st;
297    }

299    @SuppressWarnings("rawtypes")
300    private static void retrieveStylesFromParsedValue(
301            List<Style> lst, CssStyle current, ParsedValue<?, ?> parsedValue) {
302        final Object val = parsedValue.getValue();
303        if (val instanceof ParsedValue[][]) {
304            // If ParsedValue is a layered sequence of values, resolve the lookups for each.
305
306            final ParsedValue[][] layers2 = (ParsedValue[][]) val;
307            for (ParsedValue[] layers : layers2) {
308                for (ParsedValue layer : layers) {
309                    if (layer == null) {
310                        continue;
311                    }
312                    retrieveStylesFromParsedValue(lst, current, layer);
313                }
314            }
315        } else if (val instanceof ParsedValue[]) {
316            // If ParsedValue is a sequence of values, resolve the lookups for each.
317            final ParsedValue[] layers = (ParsedValue[]) val;
318            for (ParsedValue layer : layers) {
319                if (layer == null) {
320                    continue;
321                }
322                retrieveStylesFromParsedValue(lst, current, layer);
323            }
324        } else {
325            if (val instanceof String) {
326                String value = (String) val;
327                for (Style info : lst) {
328                    if (value.equals(info.getDeclaration().getProperty())) {
329                        // Ok matching Style
330                        CssStyle cssStyle = retrieveStyle(lst, info);
331                        current.getLookupChain().add(cssStyle);
332                    }
333                }
334            }
335        }
336    }

338    protected static List<CssStyle> getNotAppliedStyles(
339            List<Style> appliedStyles, Node node, CssMetaData<?, ?> cssMeta) {
340        List<CssStyle> ret = new ArrayList<>();
341
342        List<Style> allStyles = Deprecation.getMatchingStyles(cssMeta, node);
343//        System.out.println("===========================");
344//        System.out.println("getNotAppliedStyles() called!");
345//        System.out.println("===========================");
346//        System.out.println("\n\n");
347//        printStyles(allStyles);
348        List<Style> matchingStyles = removeUserAgentStyles(allStyles);
349        List<Style> notApplied = new ArrayList<>();
350        for (Style style : matchingStyles) {
351            if (!appliedStyles.contains(style)) {
352                notApplied.add(style);
353            }
354        }
355        for (Style style : notApplied) {
356            if (style.getDeclaration().getProperty().equals(cssMeta.getProperty())) {
357                // We need to retrieve from allStyles, in case a lookup is shared by appliedStyles and notApplied
358                CssStyle cssStyle = retrieveStyle(matchingStyles, style);
359                ret.add(cssStyle);
360            }
361        }
362        return ret;
363    }

365    protected static List<Style> removeUserAgentStyles(List<Style> allStyles) {
366        // With SB 2, we apply explicitly Modena/Caspian theme css on user scene graph.
367        // The rules that appear with an AUTHOR origin has already been considered as USER_AGENT.
368        // So when an internal css method (such as impl_getMatchingStyles()) is called,
369        // we need here to remove all USER_AGENT styles, to avoid doublons.
370        List<Style> matchingStyles = new ArrayList<>();
371        for (Style style : allStyles) {
372            if (!(style.getDeclaration().getRule().getOrigin() == StyleOrigin.USER_AGENT)) {
373                matchingStyles.add(style);
374            }
375        }
376        return matchingStyles;
377    }

395    public static abstract class PropertyState implements Comparable<PropertyState> {
396
397        protected PropertyState(String cssValue) {
398            this.cssValue = cssValue;
399        }
400        private final List<CssStyle> notAppliedStyles = new ArrayList<>();
401        private final List<PropertyState> lst = new ArrayList<>();
402        private final String cssValue;
403
404        public abstract String getCssProperty();
405
406        public abstract Object getFxValue();
407
408        public String getCssValue() {
409            return cssValue;
410        }
411
412        public List<PropertyState> getSubProperties() {
413            return lst;
414        }
415
416        public List<CssStyle> getNotAppliedStyles() {
417            return notAppliedStyles;
418        }
419
420        @Override
421        public int compareTo(PropertyState t) {
422            PropertyState ps = t;
423            return getCssProperty().compareTo(ps.getCssProperty());
424        }
425
426        @Override
427        public boolean equals(Object obj) {
428            if (obj == null) {
429                return false;
430            }
431            if (getClass() != obj.getClass()) {
432                return false;
433            }
434            PropertyState ps = (PropertyState) obj;
435            return getCssProperty().compareTo(ps.getCssProperty()) == 0;
436        }
437
438        @Override
439        public int hashCode() {
440            int hash = 7;
441            hash = 53 * hash + Objects.hashCode(this.notAppliedStyles);
442            hash = 53 * hash + Objects.hashCode(this.lst);
443            hash = 53 * hash + Objects.hashCode(this.cssValue);
444            return hash;
445        }
446    }

507
508            private final Style style;
509            private boolean used = true;
510            private final List<CssStyle> lookupSet = new ArrayList<>();
511
512            public CssStyle(Style style) {
513                this.style = style;
514            }

517                used = false;
518            }
519
520            public boolean isUsed() {
521                return used;
522            }
523
524            public Style getStyle() {
525                return style;
526            }
527
528            public String getCssProperty() {
529                return style.getDeclaration().getProperty();
530            }
531
532            @SuppressWarnings("rawtypes")
533            public ParsedValue getParsedValue() {
534                return style.getDeclaration().getParsedValue();
535            }
536
537            public StyleOrigin getOrigin() {
538                return CssInternal.getOrigin(style);
539            }
540
541            public String getSelector() {
542                String sel = style.getSelector().toString();
543                if (sel.startsWith("*")) {//NOI18N
544                    sel = sel.substring(1);
545                }
546                return sel;
547            }
548
549            public Rule getCssRule() {
550                return style.getDeclaration().getRule();
551            }
552
553            public URL getUrl() {
554                // Workaround!
555                Rule rule = getCssRule();
556                if (rule == null) {
557                    return null;
558                } else {
559                    try {
560                        return new URL(rule.getStylesheet().getUrl());
561                    } catch (MalformedURLException ex) {
562                        System.out.println(ex.getMessage() + " " + ex);
563                        return null;
564                    }
565                }
566            }
567
568            @Override
569            public String toString() {
570                return style.toString();
571            }
572
573            public List<CssStyle> getLookupChain() {
574                return lookupSet;
575            }
576
577            @Override
578            public int hashCode() {
579                int hash = 7;
580                hash = 47 * hash + (this.style != null ? this.style.hashCode() : 0);
581                return hash;
582            }
583
584            @Override
585            public boolean equals(Object obj) {
586                if (!(obj instanceof CssStyle)) {
587                    return false;
588                }
589                CssStyle cssStyle = (CssStyle) obj;
590                return style.equals(cssStyle.style);
591            }
592        }
