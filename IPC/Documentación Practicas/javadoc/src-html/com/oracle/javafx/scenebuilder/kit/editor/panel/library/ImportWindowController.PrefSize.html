191    @Override
192    protected void cancelButtonPressed(ActionEvent e) {
193        if (exploringTask != null && exploringTask.isRunning()) {
194            exploringTask.setOnCancelled(new EventHandler<WorkerStateEvent>() {
195
196                @Override
197                public void handle(WorkerStateEvent t) {
198//                    System.out.println("Exploration of jar files has been cancelled"); //NOI18N
199                    getStage().close();
200                }
201            });
202            exploringTask.cancel(true);
203        } else {
204            getStage().close();
205        }
206        
207        exploringTask = null;
208        
209        try {
210            closeClassLoader();
211        } catch (IOException ex) {
212            showErrorDialog(ex);
213        }
214    }

216    @Override
217    protected void okButtonPressed(ActionEvent e) {
218        exploringTask = null;
219        getStage().close();
220        
221        try {
222            closeClassLoader();
223            libPanelController.copyFilesToUserLibraryDir(importFiles);
224            UserLibrary userLib = ((UserLibrary) libPanelController.getEditorController().getLibrary());
225            userLib.setFilter(getExcludedItems());
226        } catch (IOException ex) {
227            showErrorDialog(ex);
228        } finally {
229            alreadyExcludedItems.clear();
230        }
231    }

331    // This method returns a new list of File made of the union of the provided
332    // one and jar files found in the user library dir.
333    List<File> buildListOfAllFiles(List<File> importFiles) throws IOException {
334        final List<File> res = new ArrayList<>(importFiles);
335        String userLibraryDir = ((UserLibrary) libPanelController.getEditorController().getLibrary()).getPath();
336        Path userLibraryPath = new File(userLibraryDir).toPath();
337
338        Stream<Path> pathStream = Files.list(userLibraryPath);
339        Iterator<Path> pathIterator = pathStream.iterator();
340        while (pathIterator.hasNext()) {
341            Path element = pathIterator.next();
342            if (element.toString().endsWith(".jar")) { //NOI18N
343//                    System.out.println("ImportWindowController::buildListOfAllFiles: Adding " + element); //NOI18N
344                res.add(element.toFile());
345            }
346        }
347        
348        return res;
349    }

351    private void work() {
352        exploringTask = new Task<List<JarReport>>() {
353
354            @Override
355            protected List<JarReport> call() throws Exception {
356                final List<JarReport> res = new ArrayList<>();
357                numOfImportedJar = importFiles.size();
358                // The classloader takes in addition all already existing
359                // jar files stored in the user lib dir.
360                final List<File> allFiles = buildListOfAllFiles(importFiles);
361                final URLClassLoader classLoader = getClassLoaderForFiles(allFiles);
362                int index = 1;
363                for (File file : importFiles) {
364                    if (isCancelled()) {
365                        updateMessage(I18N.getString("import.work.cancelled"));
366                        break;
367                    }
368                    updateMessage(I18N.getString("import.work.exploring", file.getName()));
369//                    System.out.println("[" + index + "/" + max + "] Exploring file " + file.getName()); //NOI18N
370                    final JarExplorer explorer = new JarExplorer(Paths.get(file.getAbsolutePath()));
371                    final JarReport jarReport = explorer.explore(classLoader);
372                    res.add(jarReport);
373                    updateProgress(index, numOfImportedJar);
374                    index++;
375                }
376
377                updateProgress(numOfImportedJar, numOfImportedJar);
378                updateImportClassLoader(classLoader);
379                return res;
380            }
381        };
382
383        Thread th = new Thread(exploringTask);
384        th.setDaemon(true);
385        processingProgressIndicator.progressProperty().bind(exploringTask.progressProperty());
386
387        // We typically enter this handler when dropping jar files such as
388        // rt.jar from Java Runtime.
389        exploringTask.setOnFailed(new EventHandler<WorkerStateEvent>() {
390
391            @Override
392            public void handle(WorkerStateEvent t) {
393                // See in setOnSucceeded the explanation for the toFront below.
394                getStage().toFront();
395                updateNumOfItemsLabelAndSelectionToggleState();
396            }
397        });
398        
399        // We construct the import list only if exploration of jar files does well.
400        // If Cancel is called during the construction of the list then the import
401        // window is closed but the construction itself will continue up to the
402        // end. Do we want to make it interruptible ?
403        exploringTask.setOnSucceeded(new EventHandler<WorkerStateEvent>() {
404
405            @Override
406            public void handle(WorkerStateEvent t) {
407                assert Platform.isFxApplicationThread();
408                // This toFront() might not be necessary because import window is modal
409                // and is chained to the document window. Anyway experience showed
410                // we need it (FX 8 b106). This is suspicious, to be investigated ...
411                // But more tricky is why toFront() is called here. Mind that when toFront()
412                // is called while isShowing() returns false isn't effective: that's
413                // why toFront called at the end of controllerDidCreateStage() or
414                // controllerDidLoadContentFxml() wasn't an option. Below is the
415                // earliest place it has been proven effective, at least on my machine.
416                getStage().toFront();
417                
418                try {
419                    // We get the set of items which are already excluded prior to the current import.
420                    UserLibrary userLib = ((UserLibrary) libPanelController.getEditorController().getLibrary());
421                    alreadyExcludedItems = userLib.getFilter();
422                    
423                    List<JarReport> jarReportList = exploringTask.get(); // blocking call
424                    final Callback<ImportRow, ObservableValue<Boolean>> importRequired
425                            = new Callback<ImportRow, ObservableValue<Boolean>>() {
426                                @Override
427                                public BooleanProperty call(ImportRow row) {
428                                    return row.importRequired();
429                                }
430                            };
431                    importList.setCellFactory(CheckBoxListCell.forListView(importRequired));
432
433                    for (JarReport jarReport : jarReportList) {
434                        for (JarReportEntry e : jarReport.getEntries()) {
435                            if ((e.getStatus() == JarReportEntry.Status.OK) && e.isNode()) {
436                                boolean checked = true;
437                                // If the class we import is already listed as an excluded one
438                                // then it must appear unchecked in the list.
439                                if (alreadyExcludedItems.contains(e.getKlass().getCanonicalName())) {
440                                    checked = false;
441                                    alreadyExcludedItems.remove(e.getKlass().getCanonicalName());
442                                }
443                                final ImportRow importRow = new ImportRow(checked, e, null);
444                                importList.getItems().add(importRow);
445                                importRow.importRequired().addListener(new ChangeListener<Boolean>() {
446
447                                    @Override
448                                    public void changed(ObservableValue<? extends Boolean> ov, Boolean oldValue, Boolean newValue) {
449                                        final int numOfComponentToImport = getNumOfComponentToImport(importList);
450                                        updateOKButtonTitle(numOfComponentToImport);
451                                        updateSelectionToggleText(numOfComponentToImport);
452                                    }
453                                });
454                            }
455                        }
456                    }
457                    
458                    // Sort based on the simple class name.
459                    Collections.sort(importList.getItems(), new ImportRowComparator());
460
461                    final int numOfComponentToImport = getNumOfComponentToImport(importList);
462                    updateOKButtonTitle(numOfComponentToImport);
463                    updateSelectionToggleText(numOfComponentToImport);
464                    updateNumOfItemsLabelAndSelectionToggleState();
465                } catch (InterruptedException | ExecutionException | IOException ex) {
466                    getStage().close();
467                    showErrorDialog(ex);
468                }
469
470                unsetProcessing();
471            }
472        });
473
474        th.start();
475    }

490    void unsetProcessing() {
491        processingProgressIndicator.setVisible(false);
492        processingLabel.setVisible(false);
493        topSplitPane.setVisible(true);
494
495        importList.getSelectionModel().selectedItemProperty().addListener(new ChangeListener<ImportRow>() {
496
497            @Override
498            public void changed(ObservableValue<? extends ImportRow> ov, ImportRow t, ImportRow t1) {
499                previewGroup.getChildren().clear();
500                final String fxmlText = JarExplorer.makeFxmlText(t1.getJarReportEntry().getKlass());
501                try {
502                    FXOMDocument fxomDoc = new FXOMDocument(fxmlText, null, importClassLoader, null);
503                    zeNode = (Node) fxomDoc.getSceneGraphRoot();
504                } catch (IOException ioe) {
505                    showErrorDialog(ioe);
506                }
507                
508                // In order to get valid bounds I need to put the node into a
509                // scene and ask for full layout.
510                try {
511                    final Group visualGroup = new Group(zeNode);
512                    final Scene hiddenScene = new Scene(visualGroup);
513                    Stage hiddenStage = new Stage();
514                    hiddenStage.setScene(hiddenScene);
515                    visualGroup.applyCss();
516                    visualGroup.layout();
517                    final Bounds zeBounds = zeNode.getLayoutBounds();
518                    builtinPrefWidth = zeBounds.getWidth();
519                    builtinPrefHeight = zeBounds.getHeight();
520                    // Detach the scene !
521                    hiddenScene.setRoot(new Group());
522                    hiddenStage.close();
523                } catch (Error e) {
524                    // Experience shows that with rogue jar files (a jar file
525                    // unlikely to contain FX controls) we can enter here.
526                    // Anything better to do than setting pref size to 0 ?
527                    builtinPrefWidth = 0;
528                    builtinPrefHeight = 0;
529                }
530                
531                if (builtinPrefWidth == 0 || builtinPrefHeight == 0) {
532                    ((Region) zeNode).setPrefSize(200, 200);
533                    setSizeLabel(PrefSize.TWO_HUNDRED_BY_TWO_HUNDRED);
534                    defSizeChoice.getSelectionModel().select(2);
535                } else {
536                    setSizeLabel(PrefSize.DEFAULT);
537                    defSizeChoice.getSelectionModel().selectFirst();
538                }
539                previewGroup.getChildren().add(zeNode);
540                defSizeChoice.setDisable(false);
541                classNameLabel.setText(t1.getJarReportEntry().getKlass().getName());
542            }
543        });
544
545        // We avoid to get an empty Preview area at first.
546        if (importList.getItems().size() > 0) {
547            importList.getSelectionModel().selectFirst();
548        }
549    }

551    private URLClassLoader getClassLoaderForFiles(List<File> files) {
552        return new URLClassLoader(makeURLArrayFromFiles(files));
553    }
554
555    private URL[] makeURLArrayFromFiles(List<File> files) {
556        final URL[] result = new URL[files.size()];
557        try {
558            int index = 0;
559            for (File file : files) {
560                result[index] = file.toURI().toURL();
561                index++;
562            }
563        } catch (MalformedURLException x) {
564            throw new RuntimeException("Bug in " + getClass().getSimpleName(), x); //NOI18N
565        }
566
567        return result;
568    }

574    private int getNumOfComponentToImport(final ListView<ImportRow> list) {
575        int res = 0;
576        
577        for (final ImportRow row : list.getItems()) {
578            if (row.isImportRequired()) {
579                res++;
580            }
581        }
582        
583        return res;
584    }

586    private List<String> getExcludedItems() {
587        List<String> res = new ArrayList<>(alreadyExcludedItems);
588        
589        for (ImportRow row : importList.getItems()) {
590            if (! row.isImportRequired()) {
591                res.add(row.getCanonicalClassName());
592            }
593        }
594        return res;
595    }

598    // what action will be taken.
599    // In the most common case one or more component are selected in the list,

604    private void updateOKButtonTitle(int numOfComponentToImport) {
605        if (numOfComponentToImport == 0) {
606            if (numOfImportedJar == 1) {
607                setOKButtonTitle(I18N.getString("import.button.import.jar"));
608            } else {
609                setOKButtonTitle(I18N.getString("import.button.import.jars"));
610            }
611        } else if (numOfComponentToImport == 1) {
612            setOKButtonTitle(I18N.getString("import.button.import.component"));
613        } else {
614            setOKButtonTitle(I18N.getString("import.button.import.components"));
615        }
616    }

618    void updateNumOfItemsLabelAndSelectionToggleState() {
619        final int num = importList.getItems().size();
620        if (num == 0 || num == 1) {
621            numOfItemsLabel.setText(num + " " //NOI18N
622                    + I18N.getString("import.num.item"));
623        } else {
624            numOfItemsLabel.setText(num + " " //NOI18N
625                    + I18N.getString("import.num.items"));
626        }
627        
628        if (num >= 1) {
629            checkAllUncheckAllToggle.setDisable(false);
630        }
631    }

641    // NOTE At the end of the day some tooling in metadata will supersedes the
642    // use of this method that is only able to deal with a Region, ignoring all
643    // other cases.
644    private void updateSize(Integer choice) {
645        if (zeNode instanceof Region) {
646            PrefSize prefSize = PrefSize.values()[choice];
647            switch (prefSize) {
648                case DEFAULT:
649                    ((Region) zeNode).setPrefSize(builtinPrefWidth, builtinPrefHeight);
650                    setSizeLabel(prefSize);
651                    break;
652                case TWO_HUNDRED_BY_ONE_HUNDRED:
653                    ((Region) zeNode).setPrefSize(200, 100);
654                    setSizeLabel(prefSize);
655                    break;
656                case TWO_HUNDRED_BY_TWO_HUNDRED:
657                    ((Region) zeNode).setPrefSize(200, 200);
658                    setSizeLabel(prefSize);
659                    break;
660                default:
661                    break;
662            }
663            
664            defSizeChoice.getSelectionModel().select(choice);
665        }
666    }

668    private void setSizeLabel(PrefSize ps) {
669        switch (ps) {
670            case DEFAULT:
671                sizeLabel.setText(builtinPrefWidth + " x " + builtinPrefHeight); //NOI18N
672                break;
673            case TWO_HUNDRED_BY_ONE_HUNDRED:
674                sizeLabel.setText("200 x 100"); //NOI18N
675                break;
676            case TWO_HUNDRED_BY_TWO_HUNDRED:
677                sizeLabel.setText("200 x 200"); //NOI18N
678                break;
679            default:
680                break;
681        }
