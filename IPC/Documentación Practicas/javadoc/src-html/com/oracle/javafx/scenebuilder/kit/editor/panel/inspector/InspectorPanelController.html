330    public SectionId getExpandedSectionId() {
331        if (!isInspectorLoaded()) {
332            return null;
333        }
334        final TitledPane expandedSection = accordion.getExpandedPane();
335        final InspectorPanelController.SectionId result;
336
337        if (expandedSection == null) {
338            // all sections are collapsed
339            result = InspectorPanelController.SectionId.NONE;
340        } else if (expandedSection == propertiesTitledPane) {
341            result = InspectorPanelController.SectionId.PROPERTIES;
342        } else if (expandedSection == layoutTitledPane) {
343            result = InspectorPanelController.SectionId.LAYOUT;
344        } else if (expandedSection == codeTitledPane) {
345            result = InspectorPanelController.SectionId.CODE;
346        } else {
347            // may happen if the view mode has been changed
348            return null;
349        }
350
351        return result;
352    }

377    public void setViewMode(ViewMode mode) {
378        assert mode != null;
379        if (viewMode == mode) {
380            // no change
381            return;
382        }
383        ViewMode previousMode = viewMode;
384        viewMode = mode;
385        if (!isInspectorLoaded()) {
386            return;
387        }
388        if (previousMode == ViewMode.SECTION) {
389            previousExpandedSection = getExpandedSectionId();
390        }
391        accordion.getPanes().clear();
392        switch (mode) {
393            case SECTION:
394                accordion.getPanes().addAll(propertiesTitledPane, layoutTitledPane, codeTitledPane);
395                if (previousExpandedSection != null) {
396                    setExpandedSection(previousExpandedSection);
397                }
398                break;
399            case PROPERTY_NAME:
400            case PROPERTY_TYPE:
401                accordion.getPanes().add(allTitledPane);
402                allTitledPane.setExpanded(true);
403                rebuild();
404                break;
405            default:
406                throw new IllegalStateException("Unexpected view mode " + mode); //NOI18N
407        }
408        updateClassNameInSectionTitles();
409    }

582    private void updateInspector() {
583        if (isInspectorLoaded() && hasFxomDocument()) {
584            SelectionState newSelectionState = new SelectionState(editorController);
585            if (isInspectorStateChanged(newSelectionState) || isEditedMode()) {
586                selectionState = newSelectionState;
587                rebuild();
588            } else {
589                // we may have a property changed here.
590                selectionState = newSelectionState;
591                reset();
592            }
593        }
594    }

596    private boolean isInspectorStateChanged(SelectionState newSelectionState) {
597        // Inspector state change if one of the following is true:
598        // - selected classes change
599        // - common parent change
600        // - resolve state change
601        return (!newSelectionState.getSelectedClasses().equals(selectionState.getSelectedClasses())

671    private void buildSection(SectionId sectionId) {
672        if (sectionId == SectionId.NONE) {
673            return;
674        }
675//        System.out.println("\nBuilding section " + sectionId + " - Selection : " + selection.getEntries());
676        GridPane gridPane = getSectionContent(sectionId);
677        gridPane.getChildren().clear();
678        if (handleSelectionMessage(gridPane)) {
679            return;
680        }
681
682        // Get Metadata
683        Set<ValuePropertyMetadata> propMetaAll = getValuePropertyMetadata();
684
685        SortedMap<InspectorPath, ValuePropertyMetadata> propMetaSection = new TreeMap<>(Metadata.getMetadata().INSPECTOR_PATH_COMPARATOR);
686        assert propMetaAll != null;
687        for (ValuePropertyMetadata valuePropMeta : propMetaAll) {
688            InspectorPath inspectorPath = valuePropMeta.getInspectorPath();
689            // Check section
690            if (!isSameSection(inspectorPath.getSectionTag(), sectionId)) {
691                continue;
692            }
693            if (valuePropMeta.isStaticProperty() && !isStaticPropertyRelevant(valuePropMeta.getName())) {
694                continue;
695            }
696            if (isEditedMode()) {
697                if (!isPropertyEdited(valuePropMeta, propMetaAll)) {
698                    continue;
699                }
700            }
701            propMetaSection.put(valuePropMeta.getInspectorPath(), valuePropMeta);
702        }
703
704        if (propMetaSection.isEmpty()) {
705            displayEmptyMessage(gridPane);
706            return;
707        }
708
709        Iterator<Entry<InspectorPath, ValuePropertyMetadata>> iter = propMetaSection.entrySet().iterator();
710        String currentSubSection = ""; //NOI18N
711        int lineIndex = 0;
712        if (sectionId == SectionId.CODE) {
713            // add fx:id here, since it is not a property.
714            // It has its own sub section title
715            addSubSectionSeparator(gridPane, lineIndex, FXID_SUBSECTION_NAME);
716            lineIndex++;
717            currentSubSection = FXID_SUBSECTION_NAME;
718            lineIndex = addFxIdEditor(gridPane, lineIndex);
719        }
720        Set<PropertyName> groupProperties = new HashSet<>();
721        while (iter.hasNext()) {
722            // Loop on properties
723            Entry<InspectorPath, ValuePropertyMetadata> entry = iter.next();
724            InspectorPath inspectorPath = entry.getKey();
725            ValuePropertyMetadata propMeta = entry.getValue();
726            String newSubSection = inspectorPath.getSubSectionTag();
727//            System.out.println(inspectorPath.getSectionTag() + " - " + newSubSection + " - " + propMeta.getName());
728            if (!currentSubSection.equalsIgnoreCase(newSubSection)) {
729                addSubSectionSeparator(gridPane, lineIndex, newSubSection);
730                lineIndex++;
731                currentSubSection = newSubSection;
732            }
733            if (isGroupedProperty(propMeta.getName())) {
734                // Several properties are grouped in a single editor (e.g. AnchorPane constraints)
735                if (groupProperties.contains(propMeta.getName())) {
736                    continue;
737                }
738                PropertiesEditor propertiesEditor
739                        = getInitializedPropertiesEditor(propMeta.getName(), propMetaSection.values(), groupProperties);
740                if (propertiesEditor == null) {
741                    continue;
742                }
743                lineIndex = addInGridPane(gridPane, propertiesEditor, lineIndex);
744            } else {
745                lineIndex = addInGridPane(gridPane, getInitializedPropertyEditor(propMeta), lineIndex);
746            }
747        }
748    }

838    private ValuePropertyMetadata[] getGroupedPropertiesMetadata(PropertyName groupedPropName,
839            Collection<ValuePropertyMetadata> propMetas, Set<PropertyName> groupProperties) {
840        // For now, the SB metadata does NOT include this grouping information.
841        // Since we have for now only AnchorPane constraints properties in this case,
842        // this is handled at the inspector level.
843        // We may include this in the metadata in the future if we have a sigificant number
844        // of properties in this case (i.e. if we plan to implement editors for rotateX/Y/Z,
845        // min/max/prefWidth, etc...)
846
847        //
848        // AnchorPane anchors only for now
849        //
850        assert isAnchorConstraintsProp(groupedPropName);
851        int anchorsNb = 4;
852        ArrayList<ValuePropertyMetadata> propMetaGroup = new ArrayList<>();
853        // Create an empty list, to be able to set the entries at the right index.
854        for (int ii = 0; ii < anchorsNb; ii++) {
855            propMetaGroup.add(null);
856        }
857
858        // Loop on properties to find anchors properties
859        for (ValuePropertyMetadata propMeta : propMetas) {
860            PropertyName propName = propMeta.getName();
861            if (!isAnchorConstraintsProp(propName)) {
862                continue;
863            }
864            groupProperties.add(propName);
865            switch (propName.toString()) {
866                case Editor.topAnchorPropName:
867                    propMetaGroup.set(0, propMeta);
868                    break;
869                case Editor.rightAnchorPropName:
870                    propMetaGroup.set(1, propMeta);
871                    break;
872                case Editor.bottomAnchorPropName:
873                    propMetaGroup.set(2, propMeta);
874                    break;
875                case Editor.leftAnchorPropName:
876                    propMetaGroup.set(3, propMeta);
877                    break;
878                default:
879                    assert false;
880            }
881        }
882        return propMetaGroup.toArray(new ValuePropertyMetadata[propMetaGroup.size()]);
883    }

913    private void buildFlatContent(GridPane gridPane) {
914//        System.out.println("\nBuilding Flat panel" + " - Selection : " + selection.getEntries());
915        gridPane.getChildren().clear();
916        gridPane.getRowConstraints().clear();
917        if (handleSelectionMessage(gridPane)) {
918            return;
919        }
920        if (isSearch(gridPane) && !hasSearchPattern()) {
921            addMessage(gridPane, I18N.getString("inspector.message.searchpattern.empty"));
922            return;
923        }
924        boolean isOrderdByType = viewMode == ViewMode.PROPERTY_TYPE;
925
926        // Get Metadata
927        Set<ValuePropertyMetadata> propMetadatas = getValuePropertyMetadata();
928        if (propMetadatas.isEmpty()) {
929            addMessage(gridPane, I18N.getString("inspector.message.no.properties"));
930            return;
931        }
932        List<ValuePropertyMetadata> propMetadataList = Arrays.asList(propMetadatas.toArray(new ValuePropertyMetadata[propMetadatas.size()]));
933        if (isOrderdByType) {
934            Collections.sort(propMetadataList, new ValuePropertyMetadataClassComparator());
935        } else {
936            Collections.sort(propMetadataList, new ValuePropertyMetadataNameComparator());
937        }
938
939        List<ValuePropertyMetadata> orderedPropMetadatas = new ArrayList<>();
940        for (ValuePropertyMetadata valuePropMeta : propMetadataList) {
941            if (isSearch(gridPane) && !isSearchPatternMatch(valuePropMeta)) {
942                continue;
943            }
944            if (valuePropMeta.isStaticProperty() && !isStaticPropertyRelevant(valuePropMeta.getName())) {
945                continue;
946            }
947            if (isEditedMode()) {
948                if (!isPropertyEdited(valuePropMeta, propMetadataList)) {
949                    continue;
950                }
951            }
952            orderedPropMetadatas.add(valuePropMeta);
953        }
954
955        if (orderedPropMetadatas.isEmpty()) {
956            displayEmptyMessage(gridPane);
957            return;
958        }
959
960        int lineIndex = 0;
961        Set<PropertyName> groupProperties = new HashSet<>();
962        for (ValuePropertyMetadata propMeta : orderedPropMetadatas) {
963            if (isGroupedProperty(propMeta.getName())) {
964                if (groupProperties.contains(propMeta.getName())) {
965                    continue;
966                }
967                // Several properties are grouped in a single editor (e.g. AnchorPane constraints)
968                PropertiesEditor propertiesEditor
969                        = getInitializedPropertiesEditor(propMeta.getName(), new HashSet<>(orderedPropMetadatas), groupProperties);
970                if (propertiesEditor == null) {
971                    continue;
972                }
973                lineIndex = addInGridPane(gridPane, propertiesEditor, lineIndex);
974            } else {
975                lineIndex = addInGridPane(gridPane, getInitializedPropertyEditor(propMeta), lineIndex);
976            }
977        }
978    }

1027    private int addInGridPane(GridPane gridPane, Editor editor, int lineIndex) {
1028        RowConstraints row1Constraints = new RowConstraints();
1029        LayoutFormat editorLayout;
1030        HBox propNameNode;
1031        String propNameText;

1033            propNameNode = ((PropertyEditor) editor).getPropNameNode();
1034            propNameText = ((PropertyEditor) editor).getPropertyNameText();
1035            editorLayout = ((PropertyEditor) editor).getLayoutFormat();
1036        } else {

1039            propNameText = ((PropertiesEditor) editor).getPropertyNameText();

1041                editorLayout = LayoutFormat.SIMPLE_LINE_NO_NAME;
1042            } else {

1048        // For SQE tests

1066            if (editorLayout != LayoutFormat.SIMPLE_LINE_NO_NAME) {

1068                if (editorLayout == LayoutFormat.SIMPLE_LINE_CENTERED) {
1069                    // Property name, valued editor and cog menu are aligned, centered.
1070                    propNameNode.setAlignment(Pos.CENTER_LEFT);
1071                } else if (editorLayout == LayoutFormat.SIMPLE_LINE_TOP) {
1072                    // Property name, valued editor and cog menu are aligned on top.
1073                    propNameNode.setAlignment(Pos.TOP_LEFT);
1074                    row1Constraints.setValignment(VPos.TOP);
1075                } else if (editorLayout == LayoutFormat.SIMPLE_LINE_BOTTOM) {

1077                    propNameNode.setAlignment(Pos.BOTTOM_LEFT);
1078                    row1Constraints.setValignment(VPos.BOTTOM);

1088            }
1089        }
1090
1091        // Add cog menu
1092        gridPane.add(menu, 2, lineIndex);
1093
1094        lineIndex++;
1095        return lineIndex;
1096    }

1155
1156                        @Override

1158                            // requestSessionEnd
1159                            if (propertyEditor.getCommitListener() != null) {
1160                                propertyEditor.getCommitListener().handle(null);

1174                    // Editing session completed
1175                    if (editorController.isTextEditingSessionOnGoing()) {
1176//                        System.out.println("textEditingSessionDidEnd() called.");
1177                        editorController.textEditingSessionDidEnd();
1178                        if (propertyEditor.getCommitListener() != null) {

1500    private int getMax(IntegerPropertyMetadata integerPropMeta) {
1501        String propNameStr = integerPropMeta.getName().getName();
1502        if (propNameStr.contains("columnIndex") || propNameStr.contains("columnSpan")) { //NOI18N
1503            GridPane gridPane = getGridPane(propNameStr);
1504            if (gridPane == null) {
1505                // multi-selection from different GridPanes: not supported for now
1506                return getMin(integerPropMeta);
1507            }
1508            int nbColumns = Deprecation.getGridPaneColumnCount(gridPane);
1509            if (propNameStr.contains("columnIndex")) {//NOI18N
1510                // index start to 0
1511                return nbColumns - 1;
1512            }
1513            if (propNameStr.contains("columnSpan")) {//NOI18N
1514                int maxIndex = getSpanPropertyMaxIndex(propNameStr);
1515                return nbColumns - maxIndex;
1516            }
1517        }
1518        if (propNameStr.contains("rowIndex") || propNameStr.contains("rowSpan")) { //NOI18N
1519            GridPane gridPane = getGridPane(propNameStr);
1520            if (gridPane == null) {
1521                // multi-selection from different GridPanes: not supported for now
1522                return getMin(integerPropMeta);
1523            }
1524            int nbRow = Deprecation.getGridPaneRowCount(gridPane);
1525            if (propNameStr.contains("rowIndex")) {//NOI18N
1526                // index start to 0
1527                return nbRow - 1;
1528            }
1529            if (propNameStr.contains("rowSpan")) {//NOI18N
1530                int maxIndex = getSpanPropertyMaxIndex(propNameStr);
1531                return nbRow - maxIndex;
1532            }
1533        }
1534        return Integer.MAX_VALUE;
1535    }

2053     *   - the selected instances, 
2054     *   - the selected classes,
2055     *   - the common parent for the selected instances (if any),

2074            // New selection: initializePopupContent all the selection variables
2075
2076            selectedInstances.clear();

2078                final ObjectSelectionGroup osg = (ObjectSelectionGroup) selection.getGroup();

2080                    if (obj instanceof FXOMInstance) {

2090                        setExpandedSection(SectionId.LAYOUT);
2091                    }
2092                }
2093            }
2094
2095            selectedClasses.clear();
2096            for (FXOMInstance instance : selectedInstances) {
2097                if (instance.getDeclaredClass() != null) { // null means unresolved instance
2098                    selectedClasses.add(instance.getDeclaredClass());
2099                }
2100            }
2101
2102            commonParentClass = null;
2103            for (FXOMInstance instance : selectedInstances) {
2104                if (commonParentClass == null) {
2105                    // first instance
2106                    commonParentClass = getParentClass(instance);
2107                } else {
2108                    if (getParentClass(instance) != commonParentClass) {
2109                        commonParentClass = null;
2110                        break;
2111                    }
2112                }
2113            }
2114
2115            commonParentObject = null;
2116            for (FXOMInstance instance : selectedInstances) {
2117                if (commonParentObject == null) {
2118                    // first instance
2119                    commonParentObject = instance.getParentObject();

2121                    if (instance.getParentObject() != commonParentObject) {
2122                        commonParentObject = null;

2124                    }
2125                }
2126            }
2127

2129            for (FXOMInstance instance : selectedInstances) {
2130                if (instance.getSceneGraphObject() == null) {
2131                    unresolvedInstances.add(instance);
2132                }
2133            }

2182            if (ed instanceof PropertyEditor) {

2184                    editor = (PropertyEditor) ed;

2187        }
2188        if (editor == null) {
2189            // editor not found

2191        }

2194
2195        final Node valueEditorNode = editorToFocus.getValueEditor();

2197        ScrollPane sp = null;
2198        Node node = valueEditorNode.getParent();
2199        while (node != null) {
2200            if (node instanceof ScrollPane) {
2201                sp = (ScrollPane) node;
2202                break;

2204            node = node.getParent();

2206        if (sp == null) {
2207            return;
2208        }
2209

2211        final ScrollPane scrollPane = sp;
2212        double editorHeight = valueEditorNode.getLayoutBounds().getHeight();
2213        final Point2D pt = scrollPane.getContent().sceneToLocal(valueEditorNode.localToScene(0, 0));
2214        // viewport height
2215        double vpHeight = scrollPane.getViewportBounds().getHeight();
2216        // Position of the editor in the scrollPane content
2217        double selY = pt.getY();

2219        double contentHeight = scrollPane.getContent().getLayoutBounds().getHeight();
2220        // Position of the middle point of the scrollPane content
2221        double contentMiddle = contentHeight / 2;
2222        // Manage the editor height depending on its position
2223        if (selY > contentMiddle) {
2224            selY += editorHeight;
2225        } else {
2226            selY -= editorHeight;
2227        }
2228        // Compute the move to apply to position the editor on the middle of the scrollPane content
2229        double moveContent = selY - contentMiddle;
2230        // Size ratio between scrollPane content and viewport
2231        double vpRatio = contentHeight / vpHeight;
2232        // Move to apply to the editor to position it in the middle of the viewport
2233        double moveVp = moveContent / vpRatio;
2234        // Position of the editor in the viewport
2235        double selYVp = (vpHeight / 2) + moveVp;
2236        // Position in percent
2237        double scrollPos = selYVp / vpHeight;
