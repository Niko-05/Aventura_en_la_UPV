254    private void addDefaultItem(Class<?> componentClass, String section, String... qualifiers) {
255        final String name = componentClass.getSimpleName();
256        String nameWithQualifier = name;
257        if (qualifiers.length > 0) {
258            nameWithQualifier += qualifiers[0];
259        }
260        final String fxmlText = makeFxmlText(componentClass);
261        addItem(nameWithQualifier, fxmlText, section, name);
262    }

273    private void addRegionItem200x200(Class<? extends Region> componentClass, String section, String qualifier, String iconName) {
274        final String name = componentClass.getSimpleName();
275        String nameWithQualifier = name;
276        if (qualifier != null) {
277            nameWithQualifier += qualifier;
278        }
279        final String fxmlText = makeRegionFxmlText(componentClass, 200.0, 200.0);
280        String theIconName = name;
281        if (iconName != null) {
282            theIconName = iconName;
283        }
284        addItem(nameWithQualifier, fxmlText, section, theIconName);
285    }

318    private void addCustomizedItem(Class<?> componentClass, String section, 
319            String fxmlBaseName, String iconName, String qualifier) {
320        String nameWithQualifier = componentClass.getSimpleName();
321        if (qualifier != null) {
322            nameWithQualifier += qualifier;
323        }
324        final String fxmlText = readCustomizedFxmlText(fxmlBaseName);
325        assert fxmlText != null;
326        addItem(nameWithQualifier, fxmlText, section, iconName);
327    }

330    private void addItem(String name, String fxmlText, String section, String iconName) {
331        final URL iconURL = ImageUtils.getNodeIconURL(iconName + ".png"); //NOI18N
332        final LibraryItem item = new LibraryItem(name, section, fxmlText, iconURL, this);
333        itemsProperty.add(item);
334    }

337    private static String makeFxmlText(Class<?> componentClass) {
338        final StringBuilder sb = new StringBuilder();
339        
340        /*
341         * <?xml version="1.0" encoding="UTF-8"?> //NOI18N
342         * 
343         * <?import a.b.C?>
344         * 
345         * <C/>
346         */
347        
348        sb.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"); //NOI18N
349        
350        sb.append("<?import "); //NOI18N
351        sb.append(componentClass.getCanonicalName());
352        sb.append("?>"); //NOI18N
353        sb.append("<"); //NOI18N
354        sb.append(componentClass.getSimpleName());
355        sb.append("/>\n"); //NOI18N
356        
357        return sb.toString();
358    }
359    
360    private static String makeRegionFxmlText(Class<? extends Region> componentClass, 
361            double pw, double ph) {
362        final StringBuilder sb = new StringBuilder();
363        
364        /*
365         * <?xml version="1.0" encoding="UTF-8"?> //NOI18N
366         * 
367         * <?import a.b.C?>
368         * 
369         * <C prefWidth="pw" prefHeight="ph"/> //NOI18N
370         */
371        
372        sb.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"); //NOI18N
373        
374        sb.append("<?import "); //NOI18N
375        sb.append(componentClass.getCanonicalName());
376        sb.append("?>"); //NOI18N
377        sb.append("<"); //NOI18N
378        sb.append(componentClass.getSimpleName());
379        if (pw != Region.USE_COMPUTED_SIZE) {
380            sb.append(" prefWidth=\""); //NOI18N
381            sb.append(pw);
382            sb.append("\""); //NOI18N
383        }
384        if (ph != Region.USE_COMPUTED_SIZE) {
385            sb.append(" prefHeight=\""); //NOI18N
386            sb.append(ph);
387            sb.append("\""); //NOI18N
388        }
389        sb.append(" />\n"); //NOI18N
390        
391        return sb.toString();
392    }

395    private String readCustomizedFxmlText(String fxmlBaseName) {
396        
397        final StringBuilder fxmlPath = new StringBuilder();
398        fxmlPath.append("builtin/"); //NOI18N
399        fxmlPath.append(fxmlBaseName);
400        fxmlPath.append(".fxml"); //NOI18N
401        
402        final URL fxmlURL = BuiltinLibrary.class.getResource(fxmlPath.toString());
403        assert fxmlURL != null : "fxmlBaseName=" + fxmlBaseName; //NOI18N
404        final String result;
405        
406        try {
407            result = FXOMDocument.readContentFromURL(fxmlURL);
408        } catch(IOException x) {
409            throw new IllegalStateException("Bug in " + getClass().getSimpleName(), x); //NOI18N
410        }
411        
412        return result;
413    }
