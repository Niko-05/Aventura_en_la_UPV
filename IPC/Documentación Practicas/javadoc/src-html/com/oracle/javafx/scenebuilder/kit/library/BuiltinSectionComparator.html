032package com.oracle.javafx.scenebuilder.kit.library;
033
034import java.util.ArrayList;
035import java.util.Comparator;
036import java.util.List;
037
038/**
039 *
040 * @treatAsPrivate
041 */
042public class BuiltinSectionComparator implements Comparator<String> {
043    
044    private static final List<String> orderedSections = new ArrayList<>();
045    
046    static {
047        orderedSections.add(BuiltinLibrary.TAG_CONTAINERS);
048        orderedSections.add(BuiltinLibrary.TAG_CONTROLS);
049        orderedSections.add(BuiltinLibrary.TAG_MENU);
050        orderedSections.add(BuiltinLibrary.TAG_MISCELLANEOUS);
051        orderedSections.add(BuiltinLibrary.TAG_SHAPES);
052        orderedSections.add(BuiltinLibrary.TAG_CHARTS);
053        orderedSections.add(BuiltinLibrary.TAG_3D);
054    }
055    
056    
057    /*
058     * Comparator
059     */
060
061    @Override
062    public int compare(String section1, String section2) {
063        assert section1 != null;
064        assert section2 != null;
065        
066        final int index1 = orderedSections.indexOf(section1);
067        final int index2 = orderedSections.indexOf(section2);
068        final int result;
069        
070        if ((index1 != -1) && (index2 != -1)) {
071            // section1 and section2 are both predefined names
072            result = Integer.compare(index1, index2);
073        } else if (index1 != -1) {
074            // only section1 is predefined -> goes before section2
075            result = +1;
076        } else if (index2 != -1) {
077            // only section2 is predefined -> goes before section1
078            result = -1;
079        } else {
080            // section1 and section2 are both custom
081            result = section1.compareTo(section2);
082        }
083        
084        return result;
085    }
086    
087}
