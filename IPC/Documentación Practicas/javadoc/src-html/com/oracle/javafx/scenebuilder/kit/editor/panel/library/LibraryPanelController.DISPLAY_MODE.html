276    public void setDisplayMode(DISPLAY_MODE displayMode) {
277        this.currentDisplayMode = displayMode;
278        
279        switch (displayMode) {
280            case SECTIONS:
281                libAccordion.setVisible(true);
282                libAccordion.setManaged(true);
283                noSearchResults.setVisible(false);
284                noSearchResults.setManaged(false);
285                libSearchList.setVisible(false);
286                libSearchList.setManaged(false);
287                libList.setVisible(false);
288                libList.setManaged(false);
289                break;
290            case SEARCH:
291                libAccordion.setVisible(false);
292                libAccordion.setManaged(false);
293                if (libSearchList.getItems().isEmpty()) {
294                    noSearchResults.setVisible(true);
295                    noSearchResults.setManaged(true);
296                    libSearchList.setVisible(false);
297                    libSearchList.setManaged(false);
298                } else {
299                    noSearchResults.setVisible(false);
300                    noSearchResults.setManaged(false);
301                    libSearchList.setVisible(true);
302                    libSearchList.setManaged(true);
303                }
304                libList.setVisible(false);
305                libList.setManaged(false);
306                break;
307            case LIST:
308                libAccordion.setVisible(false);
309                libAccordion.setManaged(false);
310                noSearchResults.setVisible(false);
311                noSearchResults.setManaged(false);
312                libSearchList.setVisible(false);
313                libSearchList.setManaged(false);
314                libList.setVisible(true);
315                libList.setManaged(true);
316                break;
317            default:
318                break;
319        }
320    }

405    // We need to discover the sections and the content of each one before being
406    // able to populate the panel.
407    // Each section is a TitledPane that contains a ListView of LibraryItem.
408    // The order section are listed is managed by the comparator that comes with
409    // the Library. In each section the LibraryItem are sorted with alphabetical
410    // ordering.
411    // First TitledPane is expanded.
412    private void populateLibraryPanel() {
413        // libData is backend structure for all that we put in the Accordion.
414        LinkedHashMap<String, ArrayList<LibraryItem>> libData = new LinkedHashMap<>();
415        TreeSet<String> sectionNames = new TreeSet<>(getEditorController().getLibrary().getSectionComparator());
416        List<TitledPane> panes = libAccordion.getPanes();
417        
418        searchData.clear();
419        libList.getItems().clear();
420        
421        if (getEditorController().getLibrary().getItems().size() > 0) {
422            // Construct a sorted set of all lib section names.
423            for (LibraryItem item : getEditorController().getLibrary().getItems()) {
424                sectionNames.add(item.getSection());
425            }
426
427            // Create a sorted set of lib elements for each section.
428            for (String sectionName : sectionNames) {
429                libData.put(sectionName, new ArrayList<>());
430            }
431
432            // Add each LibraryItem to the appropriate set.
433            for (LibraryItem item : getEditorController().getLibrary().getItems()) {
434                libData.get(item.getSection()).add(item);
435            }
436
437            // Parse our lib data structure and populate the Accordion accordingly.
438            for (String sectionName : sectionNames) {
439                ListView<LibraryListItem> itemsList = new ListView<>();
440                itemsList.setId(sectionName + "List"); // for QE //NOI18N
441                itemsList.setCellFactory(cb);
442                itemsList.addEventHandler(KeyEvent.KEY_RELEASED, keyEventHandler);
443                Collections.sort(libData.get(sectionName), new LibraryItemNameComparator());
444                for (LibraryItem item : libData.get(sectionName)) {
445                    itemsList.getItems().add(new LibraryListItem(item));
446                }
447                TitledPane sectionPane = new TitledPane(sectionName, itemsList);
448                sectionPane.setId(sectionName); // for QE
449                sectionPane.setAnimated(true);
450                panes.add(sectionPane);
451
452                searchData.addAll(libData.get(sectionName));
453
454                libList.getItems().add(new LibraryListItem(sectionName));
455                for (LibraryItem item : libData.get(sectionName)) {
456                    libList.getItems().add(new LibraryListItem(item));
457                }
458            }
459
460            if (libAccordion.getPanes().size() >= 1) {
461                expandPaneWithName(sectionNameToKeepOpened);
462            }
463
464            if (libSearchList.getCellFactory() == null) {
465                libSearchList.setCellFactory(cb);
466            }
467
468            if (libList.getCellFactory() == null) {
469                libList.setCellFactory(cb);
470            }
471
472            libSearchList.addEventHandler(KeyEvent.KEY_RELEASED, keyEventHandler);
473            libList.addEventHandler(KeyEvent.KEY_RELEASED, keyEventHandler);
474        }
475    }

478        String sectionName = paneName;

487                libAccordion.setExpandedPane(tp);
488            }
489        }

492    private void searchPatternDidChange() {
493        if (searchPattern == null || searchPattern.isEmpty()) {
494            currentDisplayMode = previousDisplayMode;
495        } else {
496            if (currentDisplayMode != DISPLAY_MODE.SEARCH) {
497                previousDisplayMode = currentDisplayMode;
498                currentDisplayMode = DISPLAY_MODE.SEARCH;
499            }
500        }
501        
502        // The filtering is done by ignoring case, and by retaining any item that
503        // contains the given pattern. An opened question is to filter as soon as
504        // the pattern is two or more characters long: for now we react from the
505        // first character.
506        //
507        // It can occur the whole Library is changed under the foots of SceneBuilder
508        // while filtering is on going. searchData is never null so if its
509        // content is changing then the filtering result will be inacurrate:
510        // that is acceptable.
511        //
512        if (currentDisplayMode.equals(DISPLAY_MODE.SEARCH)) {
513            libSearchList.getItems().clear();
514            final ArrayList<LibraryItem> rawFilteredItem = new ArrayList<>();
515            for (LibraryItem item : searchData) {
516                if (item.getName().toUpperCase(Locale.ROOT).contains(searchPattern)) {
517                    rawFilteredItem.add(item);
518                }
519            }
520            Collections.sort(rawFilteredItem, new LibraryItemNameComparator());
521            for (LibraryItem item : rawFilteredItem) {
522                libSearchList.getItems().add(new LibraryListItem(item));
523            }
524            rawFilteredItem.clear();
525        }
526        
527        setDisplayMode(currentDisplayMode);
528    }

549    private void handleKeyEvent(KeyEvent e) {
550        // On ENTER we try to insert the item which is selected within the Library.
551        if (e.getCode() == KeyCode.ENTER) {
552            // This way of doing things requires the use of an @SuppressWarnings("unchecked")
553//            final LibraryItem item = ((ListView<LibraryItem>)e.getSource()).getSelectionModel().getSelectedItem();
554            // hence this other way below ...
555            Object source = e.getSource();
556            assert source instanceof ListView;
557            final ListView<?> list = (ListView<?>)source;
558            Object rawItem = list.getSelectionModel().getSelectedItem();
559            assert rawItem instanceof LibraryListItem;
560            final LibraryListItem listitem = (LibraryListItem)rawItem;
561            final LibraryItem item = listitem.getLibItem();
562             
563            if (getEditorController().canPerformInsert(item)) {
564                getEditorController().performInsert(item);
565            }
566            
567            e.consume();
568        }
569    }

571    private void startListeningToDrop() {
572        libPane.setOnDragDropped(new EventHandler<DragEvent>() {
573
574            @Override
575            public void handle(DragEvent t) {
576//                System.out.println("libPane onDragDropped");
577                AbstractDragSource dragSource = getEditorController().getDragController().getDragSource();
578                if (dragSource != null && dragSource instanceof DocumentDragSource) {
579                    processInternalImport(((DocumentDragSource)dragSource).getDraggedObjects());
580                } else {
581                    initiateImportDialog = false;
582                    jarAndFxmlFiles.clear();
583                    t.setDropCompleted(true);
584                    // Drop gesture is only valid when the Library is an instance of UserLibrary
585                    if (getEditorController().getLibrary() instanceof UserLibrary) {
586                        Dragboard db = t.getDragboard();
587                        if (db.hasFiles()) {
588                            final List<File> files = db.getFiles();
589                            for (File file : files) {
590                                // Keep only jar and fxml files
591                                if (file.isFile() && (file.getName().endsWith(".jar") || file.getName().endsWith(".fxml"))) { //NOI18N
592//                                System.out.println("libPane onDragDropped - Retaining file " + file.getName());
593                                    jarAndFxmlFiles.add(file);
594                                }
595                            }
596
597                            // The import dialog might be kept opened by the user quite
598                            // a long time.
599                            // On Mac (not on Win), after around 20 seconds of opening
600                            // time of the import dialog window the user sees a move from
601                            // the lib panel to the Finder of the file icon, as if the drag
602                            // is rejected.
603                            // In order to silence (mask ?) this issue there's:
604                            // - the delegation to setOnDragExited of the call of processImportJarFxml
605                            // so that current handler returns fast.
606                            // - the runLater in setOnDragExited, wrapped with a Timer set with a 1 second delay
607                            // Is there a way to be notified when the import dialog
608                            // can be run without interfering with the drag and drop sequence ?
609                            initiateImportDialog = true;
610                        }
611                    }
612                }
613            }
614        });
615        
616        libPane.setOnDragExited(new EventHandler<DragEvent>() {
617
618            @Override
619            public void handle(DragEvent t) {
620//                System.out.println("libPane onDragExited");
621                if (initiateImportDialog) {
622                    initiateImportDialog = false;
623                    final Timer timer = new Timer(true);
624                    final TimerTask timerTask = new TimerTask() {
625
626                        @Override
627                        public void run() {
628                            Platform.runLater(new Runnable() {
629
630                                @Override
631                                public void run() {
632                                    processImportJarFxml(jarAndFxmlFiles);
633                                }
634                            });
635                            // I don't need to use the timer later on so by
636                            // cancelling it right here I'm sure free resources
637                            // that otherwise would prevent the JVM from exiting.
638                            timer.cancel();
639                        }
640                    };
641                    timer.schedule(timerTask, 600); // milliseconds
642                }
643            }
644        });
645        
646        
647        libPane.setOnDragOver(new EventHandler<DragEvent>() {
648
649            @Override
650            public void handle(DragEvent t) {
651//                System.out.println("libPane onDragOver");
652                AbstractDragSource dragSource = getEditorController().getDragController().getDragSource();
653                Dragboard db = t.getDragboard();
654                // db has file when dragging a file from native file manager (Mac Finder, Windows Explorer, ...).
655                // dragSource is not null if the user drags something from Hierarchy or Content panel.
656                if (db.hasFiles() || dragSource != null) {
657                    t.acceptTransferModes(TransferMode.COPY);
658                }
659            }
660        });
661        
662        // This one is called only if lib is the source of the drop.
663        libPane.setOnDragDone(new EventHandler<DragEvent>() {
664
665            @Override
666            public void handle(DragEvent t) {
667                assert getEditorController().getDragController().getDragSource() != null;
668                getEditorController().getDragController().end();
669                t.getDragboard().clear();
670                t.consume();
671            }
672        });
673        
674    }

676    // An internal import is an import to the Library initiated from within
677    // SceneBuilder (from Content or Hierarchy).
678    // We stop the watching thread to avoid potential parsing of a file that
679    // would not yet be properly finalized on disk.
680    private void processInternalImport(List<FXOMObject> objects) {
681        sectionNameToKeepOpened = getExpandedSectionName();
682        setUserLibraryPathString();
683        Path libPath = Paths.get(userLibraryPathString);
684        boolean hasDependencies = false;
685        // Selection can be multiple: as soon as one has dependencies
686        // we won't import anything.
687        // Copy of the dependencies remains something to do (DTL-5879).
688        for (FXOMObject asset : objects) {
689            if (hasDependencies(asset)) {
690                hasDependencies = true;
691                break;
692            }
693        }
694        
695        if (hasDependencies) {
696            userLibraryUpdateRejected();
697        } else {
698            ((UserLibrary) getEditorController().getLibrary()).stopWatching();
699
700            try {
701                // The selection can be multiple, in which case each asset is
702                // processed separately.
703                for (FXOMObject asset : objects) {
704                    // Create an FXML layout as a String
705                    ArrayList<FXOMObject> selection = new ArrayList<>();
706                    selection.add(asset);
707                    final FXOMArchive fxomArchive = new FXOMArchive(selection);
708                    final FXOMArchive.Entry entry0 = fxomArchive.getEntries().get(0);
709                    String fxmlText = entry0.getFxmlText();
710
711                    // Write the FXML layout into a dedicated file stored in the user Library dir.
712                    // We use the tag name of the top element and append a number:
713                    // if Library dir already contains SplitPane_1.fxml and top element has
714                    // tag name SplitPane then we will create SplitPane_2.fxml and so on.
715                    // Note the tag name is most of the time identical to the Java class name, see DTL-6643.
716                    String prefix = asset.getGlueElement().getTagName();
717                    File fxmlFile = getUniqueFxmlFileName(prefix, userLibraryPathString);
718                    writeFxmlFile(fxmlFile, fxmlText, libPath);
719                }
720            } finally {
721                if (currentDisplayMode.equals(DISPLAY_MODE.SECTIONS)) {
722                    sectionNameToKeepOpened = UserLibrary.TAG_USER_DEFINED;
723                }
724                    
725                ((UserLibrary) getEditorController().getLibrary()).startWatching();
726            }
727        }
728    }

730    private void writeFxmlFile(File targetFile, String text, Path libPath) {
731        Path targetFilePath = Paths.get(targetFile.getPath());
732        createUserLibraryDir(libPath);
733
734        try {
735            // Create the new file
736            Files.createFile(targetFilePath);
737
738            // Write content of file
739            try (PrintWriter writer = new PrintWriter(targetFile, "UTF-8")) { //NOI18N
740                writer.write(text);
741            }
742        } catch (IOException ioe) {
743            final ErrorDialog errorDialog = new ErrorDialog(null);
744            errorDialog.setTitle(I18N.getString("error.file.create.title"));
745            errorDialog.setMessage(I18N.getString("error.file.create.message", targetFilePath.normalize().toString()));
746            errorDialog.setDetails(I18N.getString("error.write.details"));
747            errorDialog.setDebugInfoWithThrowable(ioe);
748            errorDialog.showAndWait();
749        }
750    }

752    private File getUniqueFxmlFileName(String prefix, String libDir) {
753        int suffix = 0;
754        File file = null;
755        while (file == null || file.exists()) {
756            suffix++;
757            file = new File(libDir + File.separator + prefix + "_" + suffix + ".fxml"); //NOI18N
758        }
759        
760        return file;
761    }

763    private void processImportJarFxml(List<File> importedFiles) {
764        if (importedFiles != null && !importedFiles.isEmpty()) {
765            sectionNameToKeepOpened = getExpandedSectionName();
766            Path libPath = Paths.get(((UserLibrary)getEditorController().getLibrary()).getPath());
767            // Create UserLibrary dir if missing
768            if (createUserLibraryDir(libPath)) {
769                final List<File> fxmlFiles = getSubsetOfFiles(".fxml", importedFiles); //NOI18N
770
771                if (!fxmlFiles.isEmpty() && enoughFreeSpaceOnDisk(fxmlFiles) && ! hasDependencies(fxmlFiles)) {
772                    copyFilesToUserLibraryDir(fxmlFiles);
773                    
774                    if (currentDisplayMode.equals(DISPLAY_MODE.SECTIONS)) {
775                        sectionNameToKeepOpened = UserLibrary.TAG_USER_DEFINED;
776                    }
777                }
778
779                final List<File> jarFiles = getSubsetOfFiles(".jar", importedFiles); //NOI18N
780                // For jar files we delegate to the import dialog.
781                if (!jarFiles.isEmpty() && enoughFreeSpaceOnDisk(jarFiles)) {
782                    // From here we know we will initiate the import dialog.
783                    // This is why we put application window on the front.
784                    // From there the import dialog window, which is application modal,
785                    // should come on top of it.
786                    final Window window = getPanelRoot().getScene().getWindow();
787                    if (window instanceof Stage) {
788                        final Stage stage = (Stage) window;
789                        stage.toFront();
790                    }
791
792                    final ImportWindowController iwc
793                            = new ImportWindowController(this, jarFiles, window);
794                    iwc.setToolStylesheet(getEditorController().getToolStylesheet());
795                    // See comment in OnDragDropped handle set in method startListeningToDrop.
796                    ButtonID userChoice = iwc.showAndWait();
797
798                    if (userChoice.equals(ButtonID.OK) && currentDisplayMode.equals(DISPLAY_MODE.SECTIONS)) {
799                        sectionNameToKeepOpened = UserLibrary.TAG_USER_DEFINED;
800                    }
801                }
802            }
803        }
804    }

836    private boolean enoughFreeSpaceOnDisk(List<File> files) {
837        long totalSize = Long.MAX_VALUE; // bytes
838        try {
839            for (File file : files) {
840                Path targetPath = Paths.get(file.getAbsolutePath());
841                totalSize += Files.size(targetPath);
842            }
843        } catch (IOException ioe) {
844            final ErrorDialog errorDialog = new ErrorDialog(null);
845            errorDialog.setTitle(I18N.getString("error.disk.space.title"));
846            errorDialog.setMessage(I18N.getString("error.disk.space.message"));
847            errorDialog.setDetails(I18N.getString("error.write.details"));
848            errorDialog.setDebugInfoWithThrowable(ioe);
849            errorDialog.showAndWait();
850        }
851        
852        final File libFile = new File(((UserLibrary)getEditorController().getLibrary()).getPath());
853        return totalSize < libFile.getFreeSpace();
854    }

856    // Each copy is done via an intermediate temporary file that is renamed if
857    // the copy goes well (for atomicity). If a copy fails we try to erase the
858    // temporary file to stick to an as clean as possible disk content.
859    // TODO fix DTL-5879 [When copying FXML files in lib dir we have to copy files which are external references as well]
860    void copyFilesToUserLibraryDir(List<File> files) {
861        int errorCount = 0;
862        IOException savedIOE = null;
863        String savedFileName = ""; //NOI18N
864        Path tempTargetPath = null;
865        setUserLibraryPathString();
866        
867        // Here we deactivate the UserLib so that it unlocks the files contained
868        // in the lib dir in the file system meaning (especially on Windows).
869        ((UserLibrary) getEditorController().getLibrary()).stopWatching();
870        
871        try {
872            for (File file : files) {
873                savedFileName = file.getName();
874                tempTargetPath = Paths.get(userLibraryPathString, file.getName() + TEMP_FILE_EXTENSION);
875                Path ultimateTargetPath = Paths.get(userLibraryPathString, file.getName());
876                Files.deleteIfExists(tempTargetPath);
877                Files.copy(file.toPath(), tempTargetPath, StandardCopyOption.REPLACE_EXISTING);
878                Files.move(tempTargetPath, ultimateTargetPath, StandardCopyOption.ATOMIC_MOVE);
879            }
880        } catch (IOException ioe) {
881            errorCount++;
882            savedIOE = ioe;
883        } finally {
884            if (tempTargetPath != null) {
885                try {
886                    Files.deleteIfExists(tempTargetPath);
887                } catch (IOException ioe) {
888                    errorCount++;
889                    savedIOE = ioe;
890                }
891            }
892        }
893
894        ((UserLibrary) getEditorController().getLibrary()).startWatching();
895
896        if (errorCount > 0) {
897            final ErrorDialog errorDialog = new ErrorDialog(null);
898            errorDialog.setTitle(I18N.getString("error.copy.title"));
899            if (errorCount == 1) {
900                errorDialog.setMessage(I18N.getString("error.copy.message.single", savedFileName, userLibraryPathString));
901                errorDialog.setDebugInfoWithThrowable(savedIOE);
902            } else {
903                errorDialog.setMessage(I18N.getString("error.copy.message.multiple", errorCount, userLibraryPathString));
904            }
905            errorDialog.setDetails(I18N.getString("error.write.details"));
906            errorDialog.showAndWait();
907        }
908    }

936    private boolean hasDependencies(List<File> fxmlFiles) {
937        boolean hasDependencies = false;
938        boolean scanWentWell = true;
939
940        for (File fxmlFile : fxmlFiles) {
941            try {
942                if (hasDependencies(fxmlFile)) {
943                    hasDependencies = true;
944                    break;
945                }
946            } catch (IOException ioe) {
947                scanWentWell = false;
948                hasDependencies = true; // not sure but better take no risk
949                final ErrorDialog errorDialog = new ErrorDialog(null);
950                errorDialog.setTitle(I18N.getString("error.import.reject.dependencies.scan.title"));
951                errorDialog.setMessage(I18N.getString("error.import.reject.dependencies.scan.message"));

953                errorDialog.setDebugInfoWithThrowable(ioe);
954                errorDialog.showAndWait();
955            }
956        }
957
958        if (hasDependencies && scanWentWell) {
959            userLibraryUpdateRejected();
960        }
961
962        return hasDependencies;
963    }

965    private boolean hasDependencies(File fxmlFile) throws IOException {
966        boolean res = false;
967        URL location;
968        
969        location = fxmlFile.toURI().toURL();
970        FXOMDocument fxomDocument =
971                new FXOMDocument(FXOMDocument.readContentFromURL(location), location,
972                        getEditorController().getFxomDocument().getClassLoader(),
973                        getEditorController().getFxomDocument().getResources());
974        res = hasDependencies(fxomDocument.getFxomRoot());
975        
976        return res;
977    }

979    private boolean hasDependencies(FXOMObject rootFxomObject) {
980        final List<Path> targetPaths = getDependenciesPaths(rootFxomObject);
981        return targetPaths.size() > 0;
982    }
983    
984    private List<Path> getDependenciesPaths(FXOMObject rootFxomObject) {
985
986        final List<Path> targetPaths = new ArrayList<>();
987
988        for (FXOMPropertyT p : rootFxomObject.collectPropertiesT()) {
989            final Path path = extractPath(p);
990            if (path != null) {
991                targetPaths.add(path);
992            }
993        }
994        
995        for (FXOMObject fxomObject : rootFxomObject.collectObjectWithSceneGraphObjectClass(URL.class)) {
996            if (fxomObject instanceof FXOMInstance) {
997                final FXOMInstance urlInstance = (FXOMInstance) fxomObject;
998                final FXOMProperty valueProperty = urlInstance.getProperties().get(valueName);
999                if (valueProperty instanceof FXOMPropertyT) {
1000                    FXOMPropertyT valuePropertyT = (FXOMPropertyT) valueProperty;
1001                    final Path path = extractPath(valuePropertyT);
1002                    if (path != null) {
1003                        targetPaths.add(path);
1004                    }
1005                } else {
1006                    assert false : "valueProperty.getName() = " + valueProperty.getName();
1007                }
1008            }
1009        }
1010
1011        return targetPaths;
1012    }

1015    private Path extractPath(FXOMPropertyT p) {
1016        Path result;
1017        
1018        final PrefixedValue pv = new PrefixedValue(p.getValue());
1019        if (pv.isPlainString()) {
1020            try {
1021                final URL url = new URL(pv.getSuffix());
1022                result = Paths.get(url.toURI());
1023            } catch(MalformedURLException|URISyntaxException x) {
1024                result = null;
1025            }
1026        } else if (pv.isDocumentRelativePath()) {
1027            final URL documentLocation = p.getFxomDocument().getLocation();
1028            if (documentLocation == null) {
1029                result = null;
1030            } else {
1031                final URL url = pv.resolveDocumentRelativePath(documentLocation);
1032                if (url == null) {
1033                    result = null;
1034                } else {
1035                    try {
1036                        result = Paths.get(url.toURI());
1037                    } catch(FileSystemNotFoundException|URISyntaxException x) {
1038                        result = null;
1039                    }
1040                }
1041            }
1042        } else if (pv.isClassLoaderRelativePath()) {
1043            final ClassLoader classLoader = p.getFxomDocument().getClassLoader();
1044            if (classLoader == null) {
1045                result = null;
1046            } else {
1047                final URL url = pv.resolveClassLoaderRelativePath(classLoader);
1048                if (url == null) {
1049                    result = null;
1050                } else {
1051                    try {
1052                        result = Paths.get(url.toURI());
1053                    } catch(URISyntaxException x) {
1054                        result = null;
1055                    }
1056                }
1057                
1058            }
1059        } else {
1060            result = null;
1061        }
1062        
1063        return result;
1064    }
