034import java.io.File;
035import java.net.URISyntaxException;
036import java.net.URL;
037import java.util.ArrayList;
038import java.util.Collections;
039import java.util.List;
040
041/**
042 *
043 * 
044 */
045public class BuiltinGlossary extends Glossary {
046    
047    public BuiltinGlossary() {
048    }
049
050    /*
051     * Glossary
052     */
053    
054    @Override
055    public List<String> queryControllerClasses(URL fxmlLocation) {
056        if (fxmlLocation == null ) {
057            return Collections.emptyList();
058        } else {
059            File fxmlFile = getFileFromURL(fxmlLocation);
060            if (! fxmlFile.exists()) {
061                // Suspicious ! May I print some warning ? or assert the file exists ?
062                return Collections.emptyList();
063            } else {
064                List<String> res = new ArrayList<>();
065                for (ControllerClass cc : ControllerClass.discoverFXMLControllerClasses(fxmlFile)) {
066                    if (! res.contains(cc.getClassName())) {
067                        res.add(cc.getClassName());
068                    }
069                }
070                Collections.sort(res);
071                return res;
072            }
073        }
074    }
075
076    @Override
077    public List<String> queryFxIds(URL fxmlLocation, String controllerClass, Class<?> targetType) {
078        // TODO fix DTL-5878
079        assert controllerClass != null;
080        if (fxmlLocation == null ) {
081            return Collections.emptyList();
082        } else {
083            File fxmlFile = getFileFromURL(fxmlLocation);
084            if (! fxmlFile.exists()) {
085                // Suspicious ! May I print some warning ? or assert the file exists ?
086                return Collections.emptyList();
087            } else {
088                List<String> res = new ArrayList<>();
089                for (ControllerClass cc : ControllerClass.discoverFXMLControllerClasses(fxmlFile)) {
090                    if (controllerClass.equals(cc.getClassName())) {
091                        res.addAll(cc.getFxIds());
092                        break;  // discoverFXMLControllerClasses may return duplicates.
093                                // The first matching class name is good enough for now.
094                    }
095                }
096                Collections.sort(res);
097                return res;
098            }
099        }
100    }
101    
102    @Override
103    public List<String> queryEventHandlers(URL fxmlLocation, String controllerClass) {
104        assert controllerClass != null;
105        if (fxmlLocation == null ) {
106            return Collections.emptyList();
107        } else {
108            File fxmlFile = getFileFromURL(fxmlLocation);
109            if (! fxmlFile.exists()) {
110                // Suspicious ! May I print some warning ? or assert the file exists ?
111                return Collections.emptyList();
112            } else {
113                List<String> res = new ArrayList<>();
114                for (ControllerClass cc : ControllerClass.discoverFXMLControllerClasses(fxmlFile)) {
115                    if (controllerClass.equals(cc.getClassName())) {
116                        res.addAll(cc.getEventHandlers());
117                        break;  // discoverFXMLControllerClasses may return duplicates.
118                                // The first matching class name is good enough for now.
119                    }
120                }
121                Collections.sort(res);
122                return res;
123            }
124        }
125    }
126    
127    // It's better to use URL.toURI than URL.getPath to feed File constructor.
128    private File getFileFromURL(URL location) {
129        File res;
130        
131        try {
132            res= new File(location.toURI());
133        } catch (URISyntaxException ex) {
134            throw new RuntimeException("Bug", ex); //NOI18N
135        }
136        
137        return res;
138    }
139}
